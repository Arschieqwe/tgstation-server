132  async Task<IActionResult> CheckReleasesAndApplyUpdate(Version newVersion, CancellationToken cancellationToken)
133  {
134  Logger.LogDebug("Looking for GitHub releases version {0}...", newVersion);
135  IEnumerable<Release> releases;
136  try
137  {
138  var gitHubClient = GetGitHubClient();
139  releases = await gitHubClient
140  .Repository
141  .Release
142  .GetAll(updatesConfiguration.GitHubRepositoryId)
143  .WithToken(cancellationToken)
144  .ConfigureAwait(false);
145  }
146  catch (RateLimitExceededException e)
147  {
148  return RateLimit(e);
149  }
150  catch (ApiException e)
151  {
152  Logger.LogWarning(e, OctokitException);
153  return StatusCode(HttpStatusCode.FailedDependency);
154  }
155 
156  releases = releases.Where(x => x.TagName.StartsWith(updatesConfiguration.GitTagPrefix, StringComparison.InvariantCulture));
157 
158  Logger.LogTrace("Release query complete!");
159 
160  foreach (var release in releases)
161  if (Version.TryParse(
162  release.TagName.Replace(
163  updatesConfiguration.GitTagPrefix, String.Empty, StringComparison.Ordinal),
164  out var version)
165  && version == newVersion)
166  {
167  var asset = release.Assets.Where(x => x.Name.Equals(updatesConfiguration.UpdatePackageAssetName, StringComparison.Ordinal)).FirstOrDefault();
168  if (asset == default)
169  continue;
170 
171  if (!serverUpdater.ApplyUpdate(version, new Uri(asset.BrowserDownloadUrl), ioManager))
172  return Conflict(new ErrorMessage(ErrorCode.ServerUpdateInProgress));
173  return Accepted(new Administration
174  {
175  WindowsHost = platformIdentifier.IsWindows,
176  NewVersion = newVersion
177  }); // gtfo of here before all the cancellation tokens fire
178  }
179 
180  return Gone();
181  }
