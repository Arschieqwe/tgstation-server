136 public async Task<IActionResult> Read(CancellationToken cancellationToken)
137 {
138 try
139 {
140 Version greatestVersion = null;
141 Uri repoUrl = null;
142 try
143 {
144 var repositoryUrlTask = gitHubService.GetUpdatesRepositoryUrl(cancellationToken);
145 var releases = await gitHubService.GetTgsReleases(cancellationToken);
146
147 foreach (var kvp in releases)
148 {
149 var version = kvp.Key;
150 var release = kvp.Value;
151 if (version.Major > 3 // Forward/backward compatible but not before TGS4
152 && (greatestVersion == null || version > greatestVersion))
153 greatestVersion = version;
154 }
155
156 repoUrl = await repositoryUrlTask;
157 }
158 catch (NotFoundException e)
159 {
160 Logger.LogWarning(e, "Not found exception while retrieving upstream repository info!");
161 }
162
163 return Json(new AdministrationResponse
164 {
165 LatestVersion = greatestVersion,
166 TrackedRepositoryUrl = repoUrl,
167 });
168 }
169 catch (RateLimitExceededException e)
170 {
171 return RateLimit(e);
172 }
173 catch (ApiException e)

175 Logger.LogWarning(e, OctokitException);

201 public async Task<IActionResult> Update([FromBody] ServerUpdateRequest model, CancellationToken cancellationToken)
202 {
203 if (model == null)
204 throw new ArgumentNullException(nameof(model));
205
206 if (model.NewVersion == null)
207 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure)
208 {
209 AdditionalData = "newVersion is required!",
210 });
211
212 if (model.NewVersion.Major < 3)
213 return BadRequest(new ErrorMessageResponse(ErrorCode.CannotChangeServerSuite));
214
215 if (!serverControl.WatchdogPresent)
216 return UnprocessableEntity(new ErrorMessageResponse(ErrorCode.MissingHostWatchdog));
217
218 try
219 {
220 var updateResult = await serverUpdateInitiator.InitiateUpdate(model.NewVersion, cancellationToken);
221 return updateResult switch
222 {
223 ServerUpdateResult.Started => Accepted(new ServerUpdateResponse
224 {
225 NewVersion = model.NewVersion,
226 }),
227 ServerUpdateResult.ReleaseMissing => Gone(),
228 ServerUpdateResult.UpdateInProgress => BadRequest(new ErrorMessageResponse(ErrorCode.ServerUpdateInProgress)),
229 ServerUpdateResult.SwarmIntegrityCheckFailed => StatusCode(HttpStatusCode.FailedDependency, new ErrorMessageResponse(ErrorCode.SwarmIntegrityCheckFailed)),
230 _ => throw new InvalidOperationException($"Unexpected ServerUpdateResult: {updateResult}"),
231 };
232 }
233 catch (RateLimitExceededException e)
234 {
235 return RateLimit(e);
236 }
237 catch (ApiException e)
238 {
239 Logger.LogWarning(e, OctokitException);
240 return StatusCode(HttpStatusCode.FailedDependency, new ErrorMessageResponse(ErrorCode.RemoteApiError)
241 {
242 AdditionalData = e.Message,
243 });
244 }
