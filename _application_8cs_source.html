400 
401  // suppress OperationCancelledExceptions, they are just aborted HTTP requests

403 
404  if (hostingEnvironment.IsDevelopment())
405  {
406  applicationBuilder.UseSwagger();
407  applicationBuilder.UseSwaggerUI(c => c.SwaggerEndpoint("/swagger/v1/swagger.json", "TGS API V4"));
408  }
409 
410  // Set up CORS based on configuration if necessary
411  Action<CorsPolicyBuilder> corsBuilder = null;
412  if (controlPanelConfiguration.AllowAnyOrigin)
413  {
414  logger.LogTrace("Access-Control-Allow-Origin: *");
415  corsBuilder = builder => builder.AllowAnyOrigin();
416  }
417  else if (controlPanelConfiguration.AllowedOrigins?.Count > 0)
418  {
419  logger.LogTrace("Access-Control-Allow-Origin: ", String.Join(',', controlPanelConfiguration.AllowedOrigins));
420  corsBuilder = builder => builder.WithOrigins(controlPanelConfiguration.AllowedOrigins.ToArray());
