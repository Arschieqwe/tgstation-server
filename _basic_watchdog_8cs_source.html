86  async Task<MonitorAction> HandleMonitorWakeup(MonitorActivationReason reason, CancellationToken cancellationToken)
87  {
88  switch (reason)
89  {
90  case MonitorActivationReason.ActiveServerCrashed:
91  string exitWord = Server.TerminationWasRequested ? "exited" : "crashed";
92  if (Server.RebootState == Watchdog.RebootState.Shutdown)
93  {
94  // the time for graceful shutdown is now
95  await Chat.SendWatchdogMessage(String.Format(CultureInfo.InvariantCulture, "Server {0}! Exiting due to graceful termination request...", exitWord), cancellationToken).ConfigureAwait(false);
96  DisposeAndNullControllers();
97  return MonitorAction.Exit;
98  }
99 
100  await Chat.SendWatchdogMessage(String.Format(CultureInfo.InvariantCulture, "Server {0}! Rebooting...", exitWord), cancellationToken).ConfigureAwait(false);
101  return MonitorAction.Restart;
102  case MonitorActivationReason.ActiveServerRebooted:
103  var rebootState = Server.RebootState;
104  gracefulRebootSetDueToNewDmb = false;
105  Server.ResetRebootState();
106 
107  switch (rebootState)
108  {
109  case Watchdog.RebootState.Normal:
110  return HandleNormalReboot();
111  case Watchdog.RebootState.Restart:
112  return MonitorAction.Restart;
113  case Watchdog.RebootState.Shutdown:
114  // graceful shutdown time
115  await Chat.SendWatchdogMessage("Active server rebooted! Exiting due to graceful termination request...", cancellationToken).ConfigureAwait(false);
116  DisposeAndNullControllers();
117  return MonitorAction.Exit;
118  default:
119  throw new InvalidOperationException($"Invalid reboot state: {rebootState}");
120  }
121 
122  case MonitorActivationReason.ActiveLaunchParametersUpdated:
123  await Server.SetRebootState(Watchdog.RebootState.Restart, cancellationToken).ConfigureAwait(false);
124  return MonitorAction.Continue;
125  case MonitorActivationReason.NewDmbAvailable:
126  await HandleNewDmbAvailable(cancellationToken).ConfigureAwait(false);
127  return MonitorAction.Continue;
128  case MonitorActivationReason.InactiveServerCrashed:
129  case MonitorActivationReason.InactiveServerRebooted:
130  case MonitorActivationReason.InactiveServerStartupComplete:
131  throw new NotSupportedException($"Unsupported activation reason: {reason}");
132  default:
133  throw new InvalidOperationException($"Invalid activation reason: {reason}");
134  }

246  {
247  Logger.LogTrace("Entered MonitorLifetimes");
248 
249  // this function is responsible for calling HandlerMonitorWakeup when necessary and manitaining the MonitorState
250  var iteration = 1;
251  for (MonitorAction nextAction = MonitorAction.Continue; nextAction != MonitorAction.Exit; ++iteration)
252  {
253  // always start out with continue
254  nextAction = MonitorAction.Continue;
255 
256  // dump some info to the logs
257  Logger.LogDebug("Iteration {0} of monitor loop", iteration);
258  try
259  {
260  Logger.LogDebug("Server Compile Job ID: {0}", Server.Dmb.CompileJob.Id);
261 
262  // load the activation tasks into local variables
263  Task activeServerLifetime = Server.Lifetime;
264  var activeServerReboot = Server.OnReboot;
265  Task activeLaunchParametersChanged = ActiveParametersUpdated.Task;
266  var newDmbAvailable = DmbFactory.OnNewerDmb;
267 
268  // cancel waiting if requested
269  var cancelTcs = new TaskCompletionSource<object>();
270  using (cancellationToken.Register(() => cancelTcs.SetCanceled()))
271  {
272  var toWaitOn = Task.WhenAny(activeServerLifetime, activeServerReboot, newDmbAvailable, cancelTcs.Task, activeLaunchParametersChanged);
273 
274  // wait for something to happen
275  await toWaitOn.ConfigureAwait(false);
276  cancellationToken.ThrowIfCancellationRequested();
277  }
278 
279  var chatTask = Task.CompletedTask;
280  using (await SemaphoreSlimContext.Lock(Semaphore, cancellationToken).ConfigureAwait(false))
281  {
282  // always run HandleMonitorWakeup from the context of the semaphore lock
283  // multiple things may have happened, handle them one at a time
284  for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
285  {
286  MonitorActivationReason activationReason = default; // this will always be assigned before being used
287 
288  // process the tasks in this order and call HandlerMonitorWakup for each
289  bool CheckActivationReason(ref Task task, MonitorActivationReason testActivationReason)
290  {
291  var taskCompleted = task?.IsCompleted == true;
292  task = null;
293  if (nextAction == MonitorAction.Skip)
294  nextAction = MonitorAction.Continue;
295  else if (taskCompleted)
296  {
297  activationReason = testActivationReason;
298  return true;
299  }
300 
301  return false;
302  }
303 
304  if (CheckActivationReason(ref activeServerLifetime, MonitorActivationReason.ActiveServerCrashed)
305  || CheckActivationReason(ref activeServerReboot, MonitorActivationReason.ActiveServerRebooted)
306  || CheckActivationReason(ref newDmbAvailable, MonitorActivationReason.NewDmbAvailable)
307  || CheckActivationReason(ref activeLaunchParametersChanged, MonitorActivationReason.ActiveLaunchParametersUpdated))
308  {
309  Logger.LogTrace("Monitor activation: {0}", activationReason);
310  nextAction = await HandleMonitorWakeup(activationReason, cancellationToken).ConfigureAwait(false);
311  }
312  else
313  moreActivationsToProcess = false;
314  }
315  }
316 
317  // full reboot required
318  if (nextAction == MonitorAction.Restart)
319  {
320  Logger.LogDebug("Next state action is to restart");
321  DisposeAndNullControllers();
322 
323  for (var retryAttempts = 1; nextAction == MonitorAction.Restart; ++retryAttempts)
324  {
325  Exception launchException = null;
326  using (await SemaphoreSlimContext.Lock(Semaphore, cancellationToken).ConfigureAwait(false))
327  try
328  {
329  // use LaunchImplNoLock without announcements or restarting the monitor
330  await LaunchImplNoLock(false, false, null, cancellationToken).ConfigureAwait(false);
331  if (Running)
332  {
333  Logger.LogDebug("Relaunch successful, resetting monitor state...");
334  break; // continue on main loop
335  }
336  }
337  catch (OperationCanceledException)
338  {
339  throw;
340  }
341  catch (Exception e)
342  {
343  launchException = e;
344  }
345 
346  await chatTask.ConfigureAwait(false);
347  if (!Running)
348  {
349  if (launchException == null)
350  Logger.LogWarning("Failed to automatically restart the watchdog! Attempt: {0}", retryAttempts);
351  else
352  Logger.LogWarning("Failed to automatically restart the watchdog! Attempt: {0}, Exception: {1}", retryAttempts, launchException);
353  var retryDelay = Math.Min(Math.Pow(2, retryAttempts), 3600); // max of one hour, increasing by a power of 2 each time
354  chatTask = Chat.SendWatchdogMessage(String.Format(CultureInfo.InvariantCulture, "Failed to restart watchdog (Attempt: {0}), retrying in {1} seconds...", retryAttempts, retryDelay), cancellationToken);
355  await Task.WhenAll(AsyncDelayer.Delay(TimeSpan.FromSeconds(retryDelay), cancellationToken), chatTask).ConfigureAwait(false);
356  }
