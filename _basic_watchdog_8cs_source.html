88  protected override async Task<MonitorAction> HandleMonitorWakeup(MonitorActivationReason reason, CancellationToken cancellationToken)
89  {
90  switch (reason)
91  {
92  case MonitorActivationReason.ActiveServerCrashed:
93  string exitWord = Server.TerminationWasRequested ? "exited" : "crashed";
94  if (Server.RebootState == Session.RebootState.Shutdown)
95  {
96  // the time for graceful shutdown is now
97  await Chat.SendWatchdogMessage(
98  String.Format(
99  CultureInfo.InvariantCulture,
100  "Server {0}! Shutting down due to graceful termination request...",
101  exitWord),
102  cancellationToken)
103  .ConfigureAwait(false);
104  return MonitorAction.Exit;
105  }
106 
107  await Chat.SendWatchdogMessage(
108  String.Format(
109  CultureInfo.InvariantCulture,
110  "Server {0}! Rebooting...",
111  exitWord),
112  cancellationToken)
113  .ConfigureAwait(false);
114  return MonitorAction.Restart;
115  case MonitorActivationReason.ActiveServerRebooted:
116  var rebootState = Server.RebootState;
117  if (gracefulRebootRequired && rebootState == Session.RebootState.Normal)
118  {
119  Logger.LogError("Watchdog reached normal reboot state with gracefulRebootRequired set!");
120  rebootState = Session.RebootState.Restart;
121  }
122 
123  gracefulRebootRequired = false;
124  Server.ResetRebootState();
125 
126  switch (rebootState)
127  {
128  case Session.RebootState.Normal:
129  return await HandleNormalReboot(cancellationToken).ConfigureAwait(false);
130  case Session.RebootState.Restart:
131  return MonitorAction.Restart;
132  case Session.RebootState.Shutdown:
133  // graceful shutdown time
134  await Chat.SendWatchdogMessage(
135  "Active server rebooted! Shutting down due to graceful termination request...",
136  cancellationToken)
137  .ConfigureAwait(false);
138  return MonitorAction.Exit;
139  default:
140  throw new InvalidOperationException($"Invalid reboot state: {rebootState}");
141  }
142 
143  case MonitorActivationReason.ActiveLaunchParametersUpdated:
144  await Server.SetRebootState(Session.RebootState.Restart, cancellationToken).ConfigureAwait(false);
145  gracefulRebootRequired = true;
146  break;
147  case MonitorActivationReason.NewDmbAvailable:
148  await HandleNewDmbAvailable(cancellationToken).ConfigureAwait(false);
149  break;
150  case MonitorActivationReason.Heartbeat:
151  default:
152  throw new InvalidOperationException($"Invalid activation reason: {reason}");
153  }
154 
155  return MonitorAction.Continue;
156  }

175  Task chatTask,
176  ReattachInformation reattachInfo,
177  CancellationToken cancellationToken)
178  {
179  // don't need a new dmb if reattaching
180  var reattachInProgress = reattachInfo != null;
181  var dmbToUse = reattachInProgress ? null : DmbFactory.LockNextDmb(1);
182 
183  // if this try catches something, both servers are killed
184  try
185  {
186  // start the alpha server task, either by launch a new process or attaching to an existing one
187  // The tasks returned are mainly for writing interop files to the directories among other things and should generally never fail
188  // The tasks pertaining to server startup times are in the ISessionControllers
189  Task<ISessionController> serverLaunchTask;
190  if (!reattachInProgress)
191  {
192  Logger.LogTrace("Initializing controller with CompileJob {0}...", dmbToUse.CompileJob.Id);
193  await BeforeApplyDmb(dmbToUse.CompileJob, cancellationToken).ConfigureAwait(false);
194  dmbToUse = await PrepServerForLaunch(dmbToUse, cancellationToken).ConfigureAwait(false);
195 
196  await chatTask.ConfigureAwait(false);
197  serverLaunchTask = SessionControllerFactory.LaunchNew(
198  dmbToUse,
199  null,
200  ActiveLaunchParameters,
201  false,
202  cancellationToken);
203  }
204  else
205  {
206  await chatTask.ConfigureAwait(false);
207  serverLaunchTask = SessionControllerFactory.Reattach(reattachInfo, cancellationToken);
208  }
209 
210  // retrieve the session controller
211  Server = await serverLaunchTask.ConfigureAwait(false);
212 
213  // possiblity of null servers due to failed reattaches
214  if (Server == null)
215  {
216  await ReattachFailure(
217  cancellationToken)
218  .ConfigureAwait(false);
219  return;
220  }
221 
222  Server.SetHighPriority();
223 
224  await CheckLaunchResult(Server, "Server", cancellationToken).ConfigureAwait(false);
225 
226  Server.EnableCustomChatCommands();
227  }
228  catch (Exception ex)

231
