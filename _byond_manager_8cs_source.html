214 async Task ReadVersion(string path)
215 {
216 var versionFile = ioManager.ConcatPath(path, VersionFileName);
217 if (!await ioManager.FileExists(versionFile, cancellationToken))
218 {
219 logger.LogInformation("Cleaning unparsable version path: {0}", ioManager.ResolvePath(path));
220 await ioManager.DeleteDirectory(path, cancellationToken); // cleanup
221 return;
222 }
223
224 var bytes = await ioManager.ReadAllBytes(versionFile, cancellationToken);
225 var text = Encoding.UTF8.GetString(bytes);
226 if (Version.TryParse(text, out var version))
227 {
228 var key = VersionKey(version, true);
229 lock (installedVersions)
230 if (!installedVersions.ContainsKey(key))
231 {
232 logger.LogDebug("Adding detected BYOND version {0}...", key);
233 installedVersions.Add(key, Task.CompletedTask);
234 return;
235 }
236 }
237
238 await ioManager.DeleteDirectory(path, cancellationToken);
239 }
240
241 await Task.WhenAll(directories.Select(x => ReadVersion(x)));
242
243 var activeVersionBytes = await activeVersionBytesTask;
244 if (activeVersionBytes != null)
245 {
246 var activeVersionString = Encoding.UTF8.GetString(activeVersionBytes);
247 bool hasRequestedActiveVersion;
248 lock (installedVersions)
249 hasRequestedActiveVersion = installedVersions.ContainsKey(activeVersionString);
250 if (hasRequestedActiveVersion && Version.TryParse(activeVersionString, out var activeVersion))
251 ActiveVersion = activeVersion.Semver();
252 else
253 {
254 logger.LogWarning("Failed to load saved active version {0}!", activeVersionString);
255 await ioManager.DeleteFile(ActiveVersionFileName, cancellationToken);
256 }
257 }

270 async Task<string> InstallVersion(Version version, Stream customVersionStream, CancellationToken cancellationToken)
271 {
272 var ourTcs = new TaskCompletionSource<object>();
273 Task inProgressTask;
274 string versionKey;
275 bool installed;
276 lock (installedVersions)
277 {
278 if (customVersionStream != null)
279 {
280 int customInstallationNumber = 1;
281 do
282 {
283 versionKey = $"{VersionKey(version, false)}.{customInstallationNumber++}";
284 }
285 while (installedVersions.ContainsKey(versionKey));
286 }
287 else
288 versionKey = VersionKey(version, true);
289
290 installed = installedVersions.TryGetValue(versionKey, out inProgressTask);
291 if (!installed)
292 installedVersions.Add(versionKey, ourTcs.Task);
293 }
294
295 if (installed)
296 using (cancellationToken.Register(() => ourTcs.SetCanceled()))
297 {
298 await Task.WhenAny(ourTcs.Task, inProgressTask);
299 cancellationToken.ThrowIfCancellationRequested();
300 return versionKey;
301 }
302
303 if (customVersionStream != null)
304 logger.LogInformation("Installing custom BYOND version as {0}...", versionKey);
305 else if (version.Build > 0)
306 throw new JobException(ErrorCode.ByondNonExistentCustomVersion);
307 else
308 logger.LogDebug("Requested BYOND version {0} not currently installed. Doing so now...", versionKey);
309
310 // okay up to us to install it then
311 try
312 {
313 await eventConsumer.HandleEvent(EventType.ByondInstallStart, new List<string> { versionKey }, cancellationToken);
314
315 var extractPath = ioManager.ResolvePath(versionKey);
316 async Task DirectoryCleanup()
317 {
318 await ioManager.DeleteDirectory(extractPath, cancellationToken);
319 await ioManager.CreateDirectory(extractPath, cancellationToken);
320 }
321
322 var directoryCleanupTask = DirectoryCleanup();
323 try
324 {
325 Stream versionZipStream;
326 Stream downloadedStream = null;
327 if (customVersionStream == null)
328 {
329 downloadedStream = await byondInstaller.DownloadVersion(version, cancellationToken);
330 versionZipStream = downloadedStream;
331 }
332 else
333 versionZipStream = customVersionStream;
334
335 using (downloadedStream)
336 {
337 await directoryCleanupTask;
338 logger.LogTrace("Extracting downloaded BYOND zip to {0}...", extractPath);
339 await ioManager.ZipToDirectory(extractPath, versionZipStream, cancellationToken);
340 }
341
342 await byondInstaller.InstallByond(extractPath, version, cancellationToken);
343
344 // make sure to do this last because this is what tells us we have a valid version in the future
345 await ioManager.WriteAllBytes(
346 ioManager.ConcatPath(versionKey, VersionFileName),
347 Encoding.UTF8.GetBytes(versionKey),
348 cancellationToken)
349 ;
350 }
351 catch (HttpRequestException e)
352 {
353 // since the user can easily provide non-exitent version numbers, we'll turn this into a JobException
354 throw new JobException(ErrorCode.ByondDownloadFail, e);
355 }
356 catch (OperationCanceledException)
357 {
358 throw;
359 }
360 catch
361 {
362 await ioManager.DeleteDirectory(versionKey, cancellationToken);
363 throw;
364 }
365
366 ourTcs.SetResult(null);
367 }
368 catch (Exception e)
369 {
370 if (!(e is OperationCanceledException))
371 await eventConsumer.HandleEvent(EventType.ByondInstallFail, new List<string> { e.Message }, cancellationToken);
372 lock (installedVersions)
373 installedVersions.Remove(versionKey);
374 ourTcs.SetException(e);
375 throw;
376 }
377
378 return versionKey;
379 }
