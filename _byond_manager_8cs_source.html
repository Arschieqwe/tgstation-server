111  async Task InstallVersion(Version version, CancellationToken cancellationToken)
112  {
113  var ourTcs = new TaskCompletionSource<object>();
114  Task inProgressTask;
115 
116  var versionKey = VersionKey(version);
117  bool installed;
118  lock (installedVersions)
119  {
120  installed = installedVersions.TryGetValue(versionKey, out inProgressTask);
121  if (!installed)
122  installedVersions.Add(versionKey, ourTcs.Task);
123  }
124 
125  if (installed)
126  using (cancellationToken.Register(() => ourTcs.SetCanceled()))
127  {
128  await Task.WhenAny(ourTcs.Task, inProgressTask).ConfigureAwait(false);
129  cancellationToken.ThrowIfCancellationRequested();
130  return;
131  }
132 
133  // okay up to us to install it then
134  try
135  {

137  var downloadTask = byondInstaller.DownloadVersion(version, cancellationToken);
138 
139  await ioManager.DeleteDirectory(versionKey, cancellationToken).ConfigureAwait(false);
140  await ioManager.CreateDirectory(versionKey, cancellationToken).ConfigureAwait(false);
141 
142  try

145  await ioManager.ZipToDirectory(versionKey, download, cancellationToken).ConfigureAwait(false);

148  // make sure to do this last because this is what tells us we have a valid version in the future
149  await ioManager.WriteAllBytes(ioManager.ConcatPath(versionKey, VersionFileName), Encoding.UTF8.GetBytes(version.ToString()), cancellationToken).ConfigureAwait(false);
150  }
151  catch (WebException e)

154  throw new JobException(String.Format(CultureInfo.InvariantCulture, "Error downloading BYOND version: {0}", e.Message));
155  }
156  catch (OperationCanceledException)

159  }
160  catch

167  }
