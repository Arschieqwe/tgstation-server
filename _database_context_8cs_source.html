353  public async Task SchemaDowngradeForServerVersion(Version version, CancellationToken cancellationToken)
354  {
355  if (version == null)
356  throw new ArgumentNullException(nameof(version));
357  if (version < new Version(4, 0))
358  throw new ArgumentOutOfRangeException(nameof(version), version, "Not a valid V4 version!");
359 
360  // Update this with new migrations as they are made
361  string targetMigration = null;
362 
363  if (DatabaseType == DatabaseType.PostgresSql && version < new Version(4, 3, 0))
364  throw new NotSupportedException("Cannot migrate below version 4.3.0 with PostgresSql!");
365 
366  if (version < new Version(4, 1, 0))
367  throw new NotSupportedException("Cannot migrate below version 4.1.0!");
368 
369  if (version < new Version(4, 2, 0))
370  targetMigration = DatabaseType == DatabaseType.Sqlite ? nameof(SLRebuild) : nameof(MSFixCascadingDelete);
371 
372  if (targetMigration == null)
373  {
374  Logger.LogDebug("No down migration required.");
375  return;
376  }
377 
378  string migrationSubstitution;
379  switch (DatabaseType)
380  {
381  case DatabaseType.SqlServer:
382  // already setup
383  migrationSubstitution = null;
384  break;
385  case DatabaseType.MySql:
386  case DatabaseType.MariaDB:
387  migrationSubstitution = "MY{0}";
388  break;
389  case DatabaseType.Sqlite:
390  migrationSubstitution = "SL{0}";
391  break;
392  case DatabaseType.PostgresSql:
393  migrationSubstitution = "PG{0}";
394  break;
395  default:
396  throw new InvalidOperationException($"Invalid DatabaseType: {DatabaseType}");
397  }
398 
399  if (migrationSubstitution != null)
400  targetMigration = String.Format(CultureInfo.InvariantCulture, migrationSubstitution, targetMigration.Substring(2));
401 
402  // even though it clearly implements it in the DatabaseFacade definition this won't work without casting (╯ಠ益ಠ)╯︵ ┻━┻
403  var dbServiceProvider = ((IInfrastructure<IServiceProvider>)Database).Instance;
404  var migrator = dbServiceProvider.GetRequiredService<IMigrator>();
405 
406  Logger.LogInformation("Migrating down to version {0}. Target: {1}", version, targetMigration);
407  try
408  {
409  await migrator.MigrateAsync(targetMigration, cancellationToken).ConfigureAwait(false);
410  }
411  catch (Exception e)
412  {
413  Logger.LogCritical("Failed to migrate! Exception: {0}", e);
414  }
