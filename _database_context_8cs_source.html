307  await Database.EnsureDeletedAsync(cancellationToken).ConfigureAwait(false);

309 
310  var migrations = await Database.GetAppliedMigrationsAsync(cancellationToken).ConfigureAwait(false);
311  var wasEmpty = !migrations.Any();

313  if (wasEmpty || (await Database.GetPendingMigrationsAsync(cancellationToken).ConfigureAwait(false)).Any())
314  {
315  Logger.LogInformation("Migrating database...");
316  await Database.MigrateAsync(cancellationToken).ConfigureAwait(false);
317  }
318  else
319  Logger.LogDebug("No migrations to apply.");
320 
321  wasEmpty |= (await Users.AsQueryable().CountAsync(cancellationToken).ConfigureAwait(false)) == 0;
322 
323  if (wasEmpty)
324  {
325  Logger.LogInformation("Seeding database...");
