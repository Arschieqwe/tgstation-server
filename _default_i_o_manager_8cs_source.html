38 static async Task NormalizeAndDelete(DirectoryInfo dir, CancellationToken cancellationToken)
39 {
40 var tasks = new List<Task>();
41
42 // check if we are a symbolic link
43 if (!dir.Attributes.HasFlag(FileAttributes.Directory) || dir.Attributes.HasFlag(FileAttributes.ReparsePoint))
44 {
45 dir.Delete();
46 return;
47 }
48
49 foreach (var subDir in dir.EnumerateDirectories())
50 {
51 cancellationToken.ThrowIfCancellationRequested();
52 tasks.Add(NormalizeAndDelete(subDir, cancellationToken));
53 }
54
55 foreach (var file in dir.EnumerateFiles())
56 {
57 cancellationToken.ThrowIfCancellationRequested();
58 file.Attributes = FileAttributes.Normal;
59 file.Delete();
60 }
61
62 await Task.WhenAll(tasks).ConfigureAwait(false);
63 cancellationToken.ThrowIfCancellationRequested();
64 dir.Delete(true);
65 }

351 IEnumerable<Task> CopyDirectoryImpl(string src, string dest, IEnumerable<string> ignore, CancellationToken cancellationToken)
352 {
353 var dir = new DirectoryInfo(src);
354 var atLeastOneSubDir = false;
355 foreach (var subDirectory in dir.EnumerateDirectories())
356 {
357 if (ignore != null && ignore.Contains(subDirectory.Name))
358 continue;
359 foreach (var copyTask in CopyDirectoryImpl(subDirectory.FullName, Path.Combine(dest, subDirectory.Name), null, cancellationToken))
360 {
361 atLeastOneSubDir = true;
362 yield return copyTask;
363 }
364 }
365
366 async Task CopyThisDirectory()
367 {
368 if (!atLeastOneSubDir)
369 await CreateDirectory(dest, cancellationToken).ConfigureAwait(false); // save on createdir calls
370
371 var tasks = new List<Task>();
372 foreach (var fileInfo in dir.EnumerateFiles())
373 {
374 if (ignore != null && ignore.Contains(fileInfo.Name))
375 return;
376 tasks.Add(CopyFile(fileInfo.FullName, Path.Combine(dest, fileInfo.Name), cancellationToken));
377 }
378
379 await Task.WhenAll(tasks).ConfigureAwait(false);
380 }
381
382 yield return CopyThisDirectory();
