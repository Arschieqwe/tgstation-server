45 static async Task NormalizeAndDelete(DirectoryInfo dir, CancellationToken cancellationToken)
46 {
47 var tasks = new List<Task>();
48
49 // check if we are a symbolic link
50 if (!dir.Attributes.HasFlag(FileAttributes.Directory) || dir.Attributes.HasFlag(FileAttributes.ReparsePoint))
51 {
52 dir.Delete();
53 return;
54 }
55
56 foreach (var subDir in dir.EnumerateDirectories())
57 {
58 cancellationToken.ThrowIfCancellationRequested();
59 tasks.Add(NormalizeAndDelete(subDir, cancellationToken));
60 }
61
62 foreach (var file in dir.EnumerateFiles())
63 {
64 cancellationToken.ThrowIfCancellationRequested();
65 file.Attributes = FileAttributes.Normal;
66 file.Delete();
67 }
68
69 await Task.WhenAll(tasks);
70 cancellationToken.ThrowIfCancellationRequested();
71 dir.Delete(true);
72 }

373 IEnumerable<Task> CopyDirectoryImpl(string src, string dest, IEnumerable<string> ignore, CancellationToken cancellationToken)
374 {
375 var dir = new DirectoryInfo(src);
376 var atLeastOneSubDir = false;
377 foreach (var subDirectory in dir.EnumerateDirectories())
378 {
379 if (ignore != null && ignore.Contains(subDirectory.Name))
380 continue;
381 foreach (var copyTask in CopyDirectoryImpl(subDirectory.FullName, Path.Combine(dest, subDirectory.Name), null, cancellationToken))
382 {
383 atLeastOneSubDir = true;
384 yield return copyTask;
385 }
386 }
387
388 async Task CopyThisDirectory()
389 {
390 if (!atLeastOneSubDir)
391 await CreateDirectory(dest, cancellationToken); // save on createdir calls
392
393 var tasks = new List<Task>();
394 foreach (var fileInfo in dir.EnumerateFiles())
395 {
396 if (ignore != null && ignore.Contains(fileInfo.Name))
397 return;
398 tasks.Add(CopyFile(fileInfo.FullName, Path.Combine(dest, fileInfo.Name), cancellationToken));
399 }
400
401 await Task.WhenAll(tasks);
402 }
403
404 yield return CopyThisDirectory();
405 }
