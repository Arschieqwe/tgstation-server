591 protected override async Task Connect(CancellationToken cancellationToken)
592 {
593 try
594 {
595 lock (connectDisconnectLock)
596 {
597 if (gatewayCts != null)
598 throw new InvalidOperationException("Discord gateway still active!");
599
600 gatewayCts = new CancellationTokenSource();
601 }
602
603 var gatewayCancellationToken = gatewayCts.Token;
604 var gatewayClient = serviceProvider.GetRequiredService<DiscordGatewayClient>();
605
606 Task<Result> localGatewayTask;
607 gatewayReadyTcs = new TaskCompletionSource<object>();
608
609 using var gatewayConnectionAbortRegistration = cancellationToken.Register(() => gatewayReadyTcs.TrySetCanceled());
610 gatewayCancellationToken.Register(() => Logger.LogTrace("Stopping gateway client..."));
611
612 // reconnects keep happening until we stop or it faults, our auto-reconnector will handle the latter
613 localGatewayTask = gatewayClient.RunAsync(gatewayCancellationToken);
614 try
615 {
616 await Task.WhenAny(gatewayReadyTcs.Task, localGatewayTask).ConfigureAwait(false);
617
618 if (localGatewayTask.IsCompleted || cancellationToken.IsCancellationRequested)
619 throw new JobException(ErrorCode.ChatCannotConnectProvider);
620
621 var userClient = serviceProvider.GetRequiredService<IDiscordRestUserAPI>();
622
623 using var localCombinedCts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken, gatewayCancellationToken);
624 var currentUserResult = await userClient.GetCurrentUserAsync(localCombinedCts.Token).ConfigureAwait(false);
625 if (!currentUserResult.IsSuccess)
626 {
627 Logger.LogWarning("Unable to retrieve current user: {0}", currentUserResult.Error.Message);
628 throw new JobException(ErrorCode.ChatCannotConnectProvider);
629 }
630
631 currentUserId = currentUserResult.Entity.ID;
632 initialUserName = currentUserResult.Entity.Username;
633 }
634 finally
635 {
636 gatewayTask = localGatewayTask;
637 }
638 }
639 catch
640 {
641 // will handle cleanup
642 // DCT: Musn't abort
643 await DisconnectImpl(default).ConfigureAwait(false);
644 throw;
645 }
646 }

649 protected override async Task DisconnectImpl(CancellationToken cancellationToken)
650 {
651 Task<Result> localGatewayTask;
652 CancellationTokenSource localGatewayCts;
653 lock (connectDisconnectLock)
654 {
655 localGatewayTask = gatewayTask;
656 localGatewayCts = gatewayCts;
657 gatewayTask = null;
658 gatewayCts = null;

660 return;

662
663 localGatewayCts.Cancel();
664 var gatewayResult = await localGatewayTask.ConfigureAwait(false);
665 if (!gatewayResult.IsSuccess)
666 Logger.LogWarning("Gateway issue: {0}", gatewayResult.Error.Message);
