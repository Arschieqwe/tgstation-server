183 public override async Task<IReadOnlyCollection<RevInfoTestMerge>> RemoveMergedTestMerges(
184 IRepository repository,
185 RepositorySettings repositorySettings,
186 RevisionInformation revisionInformation,
187 CancellationToken cancellationToken)
188 {
189 if (repository == null)
190 throw new ArgumentNullException(nameof(repository));
191 if (repositorySettings == null)
192 throw new ArgumentNullException(nameof(repositorySettings));
193 if (revisionInformation == null)
194 throw new ArgumentNullException(nameof(revisionInformation));
195
196 if (revisionInformation.ActiveTestMerges?.Any() != true)
197 {
198 Logger.LogTrace("No test merges to remove.");
199 return Array.Empty<RevInfoTestMerge>();
200 }
201
202 var gitHubClient = repositorySettings.AccessToken != null
203 ? gitHubClientFactory.CreateClient(repositorySettings.AccessToken)
204 : gitHubClientFactory.CreateClient();
205
206 var tasks = revisionInformation
207 .ActiveTestMerges
208 .Select(x => gitHubClient
209 .PullRequest
210 .Get(repository.RemoteRepositoryOwner, repository.RemoteRepositoryName, x.TestMerge.Number)
211 .WithToken(cancellationToken));
212 try
213 {
214 await Task.WhenAll(tasks);
215 }
216 catch (Exception ex) when (ex is not OperationCanceledException)
217 {
218 Logger.LogWarning(ex, "Pull requests update check failed!");
219 }
220
221 var newList = revisionInformation.ActiveTestMerges.ToList();
222
223 PullRequest lastMerged = null;
224 async Task CheckRemovePR(Task<PullRequest> task)
225 {
226 var pr = await task;
227 if (!pr.Merged)
228 return;
229
230 // We don't just assume, actually check the repo contains the merge commit.
231 if (await repository.ShaIsParent(pr.MergeCommitSha, cancellationToken))
232 {
233 if (lastMerged == null || lastMerged.MergedAt < pr.MergedAt)
234 lastMerged = pr;
235 newList.Remove(
236 newList.First(
237 potential => potential.TestMerge.Number == pr.Number));
238 }
239 }
240
241 foreach (var prTask in tasks)
242 await CheckRemovePR(prTask);
243
244 return newList;
245 }

312 async Task UpdateDeployment(
313 CompileJob compileJob,
314 string description,
315 DeploymentState deploymentState,
316 CancellationToken cancellationToken)
317 {
318 if (compileJob == null)
319 throw new ArgumentNullException(nameof(compileJob));
320
321 if (!compileJob.GitHubRepoId.HasValue || !compileJob.GitHubDeploymentId.HasValue)
322 {
323 Logger.LogTrace("Not updating deployment as it is missing a repo ID or deployment ID.");
324 return;
325 }
326
327 Logger.LogTrace("Updating deployment {gitHubDeploymentId} to {deploymentState}...", compileJob.GitHubDeploymentId.Value, deploymentState);
328
329 string gitHubAccessToken = null;
330 await databaseContextFactory.UseContext(
331 async databaseContext =>
332 gitHubAccessToken = await databaseContext
333 .RepositorySettings
334 .AsQueryable()
335 .Where(x => x.InstanceId == Metadata.Id)
336 .Select(x => x.AccessToken)
337 .FirstAsync(cancellationToken));
338
339 if (gitHubAccessToken == null)
340 {
341 Logger.LogWarning(
342 "GitHub access token disappeared during deployment, can't update to {deploymentState}!",
343 deploymentState);
344 return;
345 }
346
347 var gitHubClient = gitHubClientFactory.CreateClient(gitHubAccessToken);
348
349 await gitHubClient
350 .Repository
351 .Deployment
352 .Status

354 compileJob.GitHubRepoId.Value,
355 compileJob.GitHubDeploymentId.Value,
