352  {
353  if (parameters == null)
354  throw new ArgumentNullException(nameof(parameters));
355 
356  IBridgeHandler bridgeHandler = null;
357  for (var i = 0; bridgeHandler == null && i < 30; ++i)
358  {
359  // There's a miniscule time period where we could potentially receive a bridge request and not have the registration ready when we launch DD
360  // This is a stopgap

362  lock (bridgeHandlers)

364  delayTask = asyncDelayer.Delay(TimeSpan.FromMilliseconds(100), cancellationToken);
365 
366  await delayTask.ConfigureAwait(false);
367  }
368 
369  if (bridgeHandler == null)
370  lock (bridgeHandlers)
371  if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
372  {
373  logger.LogWarning("Recieved invalid bridge request with accees identifier: {0}", parameters.AccessIdentifier);
374  return null;
375  }
376
