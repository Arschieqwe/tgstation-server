290 public async Task OfflineInstance(Models.Instance metadata, Models.User user, CancellationToken cancellationToken)
291 {
292 if (metadata == null)
293 throw new ArgumentNullException(nameof(metadata));
294
295 using var lockContext = await SemaphoreSlimContext.Lock(instanceStateChangeSemaphore, cancellationToken);
296
297 logger.LogInformation("Offlining instance ID {instanceId}", metadata.Id);
298 InstanceContainer container;
299 lock (instances)
300 {
301 if (!instances.TryGetValue(metadata.Id.Value, out container))
302 {
303 logger.LogDebug("Not offlining removed instance {instanceId}", metadata.Id);
304 return;
305 }
306
307 instances.Remove(metadata.Id.Value);
308 }
309
310 try
311 {
312 await container.OnZeroReferences;
313
314 // we are the one responsible for cancelling his jobs
315 var tasks = new List<Task>();
316 await databaseContextFactory.UseContext(
317 async db =>
318 {
319 var jobs = await db
320 .Jobs
321 .AsQueryable()
322 .Where(x => x.Instance.Id == metadata.Id && !x.StoppedAt.HasValue)
323 .Select(x => new Models.Job
324 {
325 Id = x.Id,
326 })
327 .ToListAsync(cancellationToken);
328 foreach (var job in jobs)
329 tasks.Add(jobManager.CancelJob(job, user, true, cancellationToken));
330 });
331
332 await Task.WhenAll(tasks);
333
334 await container.Instance.StopAsync(cancellationToken);
335 }
336 finally
337 {
338 await container.Instance.DisposeAsync();
339 }

493 public async Task<BridgeResponse> ProcessBridgeRequest(BridgeParameters parameters, CancellationToken cancellationToken)
494 {
495 if (parameters == null)
496 throw new ArgumentNullException(nameof(parameters));
497
498 IBridgeHandler bridgeHandler = null;
499 for (var i = 0; bridgeHandler == null && i < 30; ++i)
500 {
501 // There's a miniscule time period where we could potentially receive a bridge request and not have the registration ready when we launch DD
502 // This is a stopgap
503 Task delayTask = Task.CompletedTask;
504 lock (bridgeHandlers)
505 if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
506 delayTask = asyncDelayer.Delay(TimeSpan.FromMilliseconds(100), cancellationToken);
507
508 await delayTask;
509 }
510
511 if (bridgeHandler == null)
512 lock (bridgeHandlers)
513 if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
514 {
515 logger.LogWarning("Recieved invalid bridge request with access identifier: {accessIdentifier}", parameters.AccessIdentifier);
516 return null;
517 }
518
519 return await bridgeHandler.ProcessBridgeRequest(parameters, cancellationToken);
520 }

576 async Task InitializeSwarm(CancellationToken cancellationToken)
577 {
578 SwarmRegistrationResult registrationResult;
579 do
580 {
581 registrationResult = await swarmServiceController.Initialize(cancellationToken);
582
583 if (registrationResult == SwarmRegistrationResult.Unauthorized)
584 throw new InvalidOperationException("Swarm private key does not match the swarm controller's!");
585
586 if (registrationResult == SwarmRegistrationResult.VersionMismatch)
587 throw new InvalidOperationException("Swarm controller's TGS version does not match our own!");
588
589 if (registrationResult != SwarmRegistrationResult.Success)
590 await asyncDelayer.Delay(TimeSpan.FromSeconds(5), cancellationToken);
591 }
592 while (registrationResult != SwarmRegistrationResult.Success && !cancellationToken.IsCancellationRequested);
593 }
