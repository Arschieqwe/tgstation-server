501  public async Task<BridgeResponse> ProcessBridgeRequest(BridgeParameters parameters, CancellationToken cancellationToken)
502  {
503  if (parameters == null)
504  throw new ArgumentNullException(nameof(parameters));
505 
506  IBridgeHandler bridgeHandler = null;
507  for (var i = 0; bridgeHandler == null && i < 30; ++i)
508  {
509  // There's a miniscule time period where we could potentially receive a bridge request and not have the registration ready when we launch DD
510  // This is a stopgap
511  Task delayTask = Task.CompletedTask;
512  lock (bridgeHandlers)
513  if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
514  delayTask = asyncDelayer.Delay(TimeSpan.FromMilliseconds(100), cancellationToken);
515 
516  await delayTask.ConfigureAwait(false);
517  }
518 
519  if (bridgeHandler == null)
520  lock (bridgeHandlers)
521  if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
522  {
523  logger.LogWarning("Recieved invalid bridge request with accees identifier: {0}", parameters.AccessIdentifier);
524  return null;
525  }
526 
527  return await bridgeHandler.ProcessBridgeRequest(parameters, cancellationToken).ConfigureAwait(false);
