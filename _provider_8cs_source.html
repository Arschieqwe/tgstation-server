252 async Task ReconnectionLoop(uint reconnectInterval, bool connectNow, CancellationToken cancellationToken)
253 {
254 do
255 {
256 try
257 {
258 if (!connectNow)
259 await Task.Delay(TimeSpan.FromMinutes(reconnectInterval), cancellationToken);
260 else
261 connectNow = false;
262 if (!Connected)
263 {
264 var job = new Job
265 {
266 Description = $"Reconnect chat bot: {ChatBot.Name}",
267 CancelRight = (ulong)ChatBotRights.WriteEnabled,
268 CancelRightsType = RightsType.ChatBots,
269 Instance = ChatBot.Instance,
270 };
271
272 await jobManager.RegisterOperation(
273 job,
274 async (core, databaseContextFactory, paramJob, progressReporter, jobCancellationToken) =>
275 {
276 try
277 {
278 if (Connected)
279 {
280 Logger.LogTrace("Disconnecting...");
281 await DisconnectImpl(jobCancellationToken);
282 }
283 else
284 Logger.LogTrace("Already disconnected not doing disconnection attempt!");
285
286 Logger.LogTrace("Connecting...");
287 await Connect(jobCancellationToken);
288 Logger.LogTrace("Connected successfully");
289 EnqueueMessage(null);
290 }
291 catch
292 {
293 initialConnectionTcs.TrySetResult();
294 throw;
295 }
296 },
297 cancellationToken);
298
299 // DCT: Always wait for the job to complete here
300 await jobManager.WaitForJobCompletion(job, null, cancellationToken, default);
301 }
302 }
303 catch (OperationCanceledException)

305 break;
306 }
307 catch (Exception e)

309 Logger.LogError(e, "Error reconnecting!");
310 }
