224 async Task ReconnectionLoop(uint reconnectInterval, bool connectNow, CancellationToken cancellationToken)
225 {
226 do
227 {
228 try
229 {
230 if (!connectNow)
231 await Task.Delay(TimeSpan.FromMinutes(reconnectInterval), cancellationToken);
232 else
233 connectNow = false;
234 if (!Connected)
235 {
236 var job = new Job
237 {
238 Description = $"Reconnect chat bot: {ChatBot.Name}",
239 CancelRight = (ulong)ChatBotRights.WriteEnabled,
240 CancelRightsType = RightsType.ChatBots,
241 Instance = ChatBot.Instance,
242 };
243
244 await jobManager.RegisterOperation(
245 job,
246 async (core, databaseContextFactory, paramJob, progressReporter, jobCancellationToken) =>
247 {
248 if (Connected)
249 {
250 Logger.LogTrace("Disconnecting...");
251 await DisconnectImpl(jobCancellationToken);
252 }
253 else
254 Logger.LogTrace("Already disconnected not doing disconnection attempt!");
255
256 Logger.LogTrace("Connecting...");
257 await Connect(jobCancellationToken);
258 Logger.LogTrace("Connected successfully");
259 EnqueueMessage(null);
260 },
261 cancellationToken)
262 ;
263
264 // DCT: Always wait for the job to complete here
265 await jobManager.WaitForJobCompletion(job, null, cancellationToken, default);
266 }
267 }
268 catch (OperationCanceledException)
269 {
270 break;
271 }
272 catch (Exception e)
273 {
274 Logger.LogError(e, "Error reconnecting!");
275 }
276 finally
277 {
278 initialConnectionTcs.TrySetResult(null);
279 }
280 }
281 while (true);
282 }
