102 public async Task<IRepository> CloneRepository(
103 Uri url,
104 string initialBranch,
105 string username,
106 string password,
107 JobProgressReporter progressReporter,
108 bool recurseSubmodules,
109 CancellationToken cancellationToken)
110 {
111 if (url == null)
112 throw new ArgumentNullException(nameof(url));
113 if (progressReporter == null)
114 throw new ArgumentNullException(nameof(progressReporter));
115
116 logger.LogInformation("Begin clone {0} (Branch: {1})", url, initialBranch);
117 lock (semaphore)
118 {
119 if (CloneInProgress)
120 throw new JobException(ErrorCode.RepoCloning);
121 CloneInProgress = true;
122 }
123
124 try
125 {
126 using (await SemaphoreSlimContext.Lock(semaphore, cancellationToken).ConfigureAwait(false))
127 {
128 logger.LogTrace("Semaphore acquired");
129 var repositoryPath = ioManager.ResolvePath();
130 if (!await ioManager.DirectoryExists(repositoryPath, cancellationToken).ConfigureAwait(false))
131 try
132 {
133 var cloneOptions = new CloneOptions
134 {
135 OnProgress = (a) => !cancellationToken.IsCancellationRequested,
136 OnTransferProgress = (a) =>
137 {
138 var percentage = 100 * (((float)a.IndexedObjects + a.ReceivedObjects) / (a.TotalObjects * 2));
139 progressReporter("Cloning", (int)percentage);
140 return !cancellationToken.IsCancellationRequested;
141 },
142 RecurseSubmodules = recurseSubmodules,
143 OnUpdateTips = (a, b, c) => !cancellationToken.IsCancellationRequested,
144 RepositoryOperationStarting = (a) => !cancellationToken.IsCancellationRequested,
145 BranchName = initialBranch,
146 CredentialsProvider = repositoryFactory.GenerateCredentialsHandler(username, password),
147 };
148
149 await repositoryFactory.Clone(
150 url,
151 cloneOptions,
152 repositoryPath,
153 cancellationToken)
154 .ConfigureAwait(false);
155 }
156 catch
157 {
158 try
159 {
160 logger.LogTrace("Deleting partially cloned repository...");
161
162 // DCT: Cancellation token is for job, operation must run regardless
163 await ioManager.DeleteDirectory(repositoryPath, default).ConfigureAwait(false);
164 }
165 catch (Exception innerException)
166 {
167 logger.LogError(innerException, "Error deleting partially cloned repository!");
168 }
169
170 throw;
171 }
172 else
173 {
174 logger.LogDebug("Repository exists, clone aborted!");
175 return null;
176 }
177 }
178
179 logger.LogInformation("Clone complete!");
180 }
181 finally
182 {
183 CloneInProgress = false;
184 }
185
186 return await LoadRepository(cancellationToken).ConfigureAwait(false);
187 }

190 public async Task<IRepository> LoadRepository(CancellationToken cancellationToken)
191 {
192 logger.LogTrace("Begin LoadRepository...");
193 lock (semaphore)
194 if (CloneInProgress)
195 throw new JobException(ErrorCode.RepoCloning);
196 await semaphore.WaitAsync(cancellationToken).ConfigureAwait(false);
197 try
198 {
199 try
200 {
201 var libGit2Repo = await repositoryFactory.CreateFromPath(ioManager.ResolvePath(), cancellationToken).ConfigureAwait(false);
202
203 return new Repository(
204 libGit2Repo,
205 commands,
206 ioManager,
207 eventConsumer,
208 repositoryFactory,
209 gitRemoteFeaturesFactory,
210 repositoryLogger,
211 () =>
212 {
213 logger.LogTrace("Releasing semaphore due to Repository disposal...");
214 semaphore.Release();
215 });
216 }
217 catch
218 {
219 semaphore.Release();
220 throw;
221 }
222 }
223 catch (RepositoryNotFoundException e)
224 {
225 logger.LogTrace(e, "Repository not found!");
226 return null;
227 }
228 }
