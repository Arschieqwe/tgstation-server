90 public async Task<ServerUpdateResult> BeginUpdate(ISwarmService swarmService, Version newVersion, CancellationToken cancellationToken)
91 {
92 if (swarmService == null)
93 throw new ArgumentNullException(nameof(swarmService));
94
95 if (newVersion == null)
96 throw new ArgumentNullException(nameof(newVersion));
97
98 if (!swarmService.ExpectedNumberOfNodesConnected)
99 return ServerUpdateResult.SwarmIntegrityCheckFailed;
100
101 logger.LogDebug("Looking for GitHub releases version {version}...", newVersion);
102 var gitHubClient = gitHubClientFactory.CreateClient();
103 var releases = await gitHubClient
104 .Repository
105 .Release
106 .GetAll(updatesConfiguration.GitHubRepositoryId)
107 .WithToken(cancellationToken);
108
109 logger.LogTrace("Received {releaseCount} total releases from GitHub", releases.Count);
110
111 var filteredReleases = releases
112 .Where(x => x.TagName.StartsWith(updatesConfiguration.GitTagPrefix, StringComparison.InvariantCulture))
113 .ToList();
114
115 logger.LogTrace(
116 "Filtered to {releaseCount} releases matching the configured tag prefix of \"{tagPrefix}\"",
117 filteredReleases.Count,
118 updatesConfiguration.GitTagPrefix);
119
120 foreach (var release in filteredReleases)
121 if (Version.TryParse(
122 release.TagName.Replace(
123 updatesConfiguration.GitTagPrefix, String.Empty, StringComparison.Ordinal),
124 out var version))
125 {
126 if (version == newVersion)
127 {
128 var asset = release.Assets.Where(x => x.Name.Equals(updatesConfiguration.UpdatePackageAssetName, StringComparison.Ordinal)).FirstOrDefault();
129 if (asset == default)
130 continue;
131
132 serverUpdateOperation = new ServerUpdateOperation
133 {
134 TargetVersion = version,
135 UpdateZipUrl = new Uri(asset.BrowserDownloadUrl),
136 SwarmService = swarmService,
137 };
138
139 try
140 {
141 if (!serverControl.TryStartUpdate(this, version))
142 return ServerUpdateResult.UpdateInProgress;
143 }
144 finally
145 {
146 serverUpdateOperation = null;
147 }
148
149 return ServerUpdateResult.Started;
150 }
151 }
152 else
153 logger.LogDebug("Unparsable release tag: {releaseTag}", release.TagName);
154
155 if (updatesConfiguration.DumpReleasesOnNotFound)
156 logger.LogInformation(
157 "Found releases:{newline}\t{releases}",
158 Environment.NewLine,
159 String.Join($"{Environment.NewLine}\t", releases.Select(x => x.TagName).OrderBy(x => x)));
160
161 return ServerUpdateResult.ReleaseMissing;
162 }

165 public async Task<bool> ExecuteUpdate(string updatePath, CancellationToken cancellationToken, CancellationToken criticalCancellationToken)
166 {
167 if (updatePath == null)
168 throw new ArgumentNullException(nameof(updatePath));
169
170 var serverUpdateOperation = this.serverUpdateOperation;
171 if (serverUpdateOperation == null)
172 throw new InvalidOperationException($"{nameof(serverUpdateOperation)} was null!");
173
174 logger.LogInformation(
175 "Updating server to version {version} ({zipUrl})...",
176 serverUpdateOperation.TargetVersion,
177 serverUpdateOperation.UpdateZipUrl);
178
179 var inMustCommitUpdate = false;
180 try
181 {
182 var updatePrepareResult = await serverUpdateOperation.SwarmService.PrepareUpdate(serverUpdateOperation.TargetVersion, cancellationToken);
183 if (!updatePrepareResult)
184 return false;
185
186 async Task TryAbort(Exception ex)
187 {
188 try
189 {
190 await serverUpdateOperation.SwarmService.AbortUpdate(cancellationToken);
191 }
192 catch (Exception e2)
193 {
194 throw new AggregateException(ex, e2);
195 }
196 }
197
198 var stagingDirectory = $"{updatePath}-stage";
199 MemoryStream updateZipData;
200 try
201 {
202 logger.LogTrace("Downloading zip package...");
203 var bearerToken = generalConfiguration.GitHubAccessToken;
204 if (String.IsNullOrWhiteSpace(bearerToken))
205 bearerToken = null;
206
207 updateZipData = await fileDownloader.DownloadFile(serverUpdateOperation.UpdateZipUrl, bearerToken, cancellationToken);
208 }
209 catch (Exception ex)
210 {
211 await TryAbort(ex);
212 throw;
213 }
214
215 try
216 {
217 try
218 {
219 using (updateZipData)
220 {
221 logger.LogTrace("Extracting zip package to {stagingDirectory}...", stagingDirectory);
222 await ioManager.DeleteDirectory(stagingDirectory, cancellationToken);
223 await ioManager.ZipToDirectory(stagingDirectory, updateZipData, cancellationToken);
224 }
225 }
226 catch (Exception ex)
227 {
228 await TryAbort(ex);
229 throw;
230 }
231
232 var updateCommitResult = await serverUpdateOperation.SwarmService.CommitUpdate(criticalCancellationToken);
233 if (updateCommitResult == SwarmCommitResult.AbortUpdate)
234 {
235 logger.LogError("Swarm distributed commit failed, not applying update!");
236 return false;
237 }
238
239 inMustCommitUpdate = updateCommitResult == SwarmCommitResult.MustCommitUpdate;
240 logger.LogTrace("Moving {stagingDirectory} to {updateDirectory}", stagingDirectory, updatePath);
241 await ioManager.MoveDirectory(stagingDirectory, updatePath, criticalCancellationToken);
242 }
243 catch (Exception e)
244 {
245 try
246 {
247 // important to not leave this directory around if possible
248 await ioManager.DeleteDirectory(stagingDirectory, default);
249 }
250 catch (Exception e2)
251 {
252 throw new AggregateException(e, e2);
253 }
254
255 throw;
256 }
257
258 return true;
259 }
260 catch (OperationCanceledException) when (!inMustCommitUpdate)
261 {
262 logger.LogInformation("Server update cancelled!");
263 }
264 catch (Exception ex) when (!inMustCommitUpdate)
265 {
266 logger.LogError(ex, "Error updating server!");
267 }
268 catch (Exception ex) when (inMustCommitUpdate)
269 {
270 logger.LogCritical(ex, "Could not complete committed swarm update!");
271 }
272
273 return false;
274 }
