515 async Task<LaunchResult> GetLaunchResult(
516 IAssemblyInformationProvider assemblyInformationProvider,
517 uint? startupTimeout,
518 bool reattached,
519 bool apiValidate)
520 {
521 var startTime = DateTimeOffset.UtcNow;
522 var useBridgeRequestForLaunchResult = !reattached && (apiValidate || DMApiAvailable);
523 var startupTask = useBridgeRequestForLaunchResult
524 ? initialBridgeRequestTcs.Task
525 : process.Startup;
526 var toAwait = Task.WhenAny(startupTask, process.Lifetime);
527
528 if (startupTimeout.HasValue)
529 toAwait = Task.WhenAny(toAwait, Task.Delay(TimeSpan.FromSeconds(startupTimeout.Value)));
530
531 Logger.LogTrace(
532 "Waiting for LaunchResult based on {launchResultCompletionCause}{possibleTimeout}...",
533 useBridgeRequestForLaunchResult ? "initial bridge request" : "process startup",
534 startupTimeout.HasValue ? $" with a timeout of {startupTimeout.Value}s" : String.Empty);
535
536 await toAwait;
537
538 var result = new LaunchResult
539 {
540 ExitCode = process.Lifetime.IsCompleted ? await process.Lifetime : null,
541 StartupTime = startupTask.IsCompleted ? (DateTimeOffset.UtcNow - startTime) : null,
542 };
543
544 Logger.LogTrace("Launch result: {launchResult}", result);
545
546 if (!result.ExitCode.HasValue && reattached && !disposed)
547 {
548 var reattachResponse = await SendCommand(
549 new TopicParameters(
550 assemblyInformationProvider.Version,
551 ReattachInformation.RuntimeInformation.ServerPort),
552 reattachTopicCts.Token);
553
554 if (reattachResponse != null)
555 {
556 if (reattachResponse?.CustomCommands != null)
557 chatTrackingContext.CustomCommands = reattachResponse.CustomCommands;
558 else if (reattachResponse != null)
559 Logger.Log(
560 CompileJob.DMApiVersion >= new Version(5, 2, 0)
561 ? LogLevel.Warning
562 : LogLevel.Debug,
563 "DMAPI Interop v{interopVersion} isn't returning the TGS custom commands list. Functionality added in v5.2.0.",
564 CompileJob.DMApiVersion.Semver());
565 }
566 }
567
568 return result;

586 async Task<BridgeResponse> ProcessBridgeCommand(BridgeParameters parameters, CancellationToken cancellationToken)
587 {
588 var response = new BridgeResponse();
589 switch (parameters.CommandType)
590 {
591 case BridgeCommandType.ChatSend:
592 if (parameters.ChatMessage == null)
593 return BridgeError("Missing chatMessage field!");
594
595 if (parameters.ChatMessage.ChannelIds == null)
596 return BridgeError("Missing channelIds field in chatMessage!");
597
598 if (parameters.ChatMessage.ChannelIds.Any(channelIdString => !UInt64.TryParse(channelIdString, out var _)))
599 return BridgeError("Invalid channelIds in chatMessage!");
600
601 if (parameters.ChatMessage.Text == null)
602 return BridgeError("Missing message field in chatMessage!");
603
604 var anyFailed = false;
605 var parsedChannels = parameters.ChatMessage.ChannelIds.Select(
606 channelString =>
607 {
608 anyFailed |= !UInt64.TryParse(channelString, out var channelId);
609 return channelId;
610 });
611
612 if (anyFailed)
613 return BridgeError("Failed to parse channelIds as U64!");
614
615 chat.QueueMessage(
616 parameters.ChatMessage,
617 parsedChannels);
618 break;
619 case BridgeCommandType.Prime:
620 var oldPrimeTcs = primeTcs;
621 primeTcs = new TaskCompletionSource();
622 oldPrimeTcs.SetResult();
623 break;
624 case BridgeCommandType.Kill:
625 Logger.LogInformation("Bridge requested process termination!");
626 TerminationWasRequested = true;
627 process.Terminate();
628 break;
629 case BridgeCommandType.PortUpdate:
630 lock (synchronizationLock)
631 {
632 if (!parameters.CurrentPort.HasValue)
633 {
635 Logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
636 return BridgeError("Missing stringified port as data parameter!");
637 }
638
639 var currentPort = parameters.CurrentPort.Value;
640 if (!nextPort.HasValue)
641 ReattachInformation.Port = parameters.CurrentPort.Value; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
642 else
643 {
644 // nextPort is ready, tell DD to switch to that
645 // if it fails it'll kill itself
646 response.NewPort = nextPort.Value;
647 ReattachInformation.Port = nextPort.Value;
648 nextPort = null;
649
650 // we'll also get here from SetPort so complete that task
651 var tmpTcs = portAssignmentTcs;
652 portAssignmentTcs = null;
653 tmpTcs.SetResult(true);
654 }
655
656 portClosedForReboot = false;
657 }
658
659 break;
660 case BridgeCommandType.Startup:
661 apiValidationStatus = ApiValidationStatus.BadValidationRequest;
662 if (parameters.Version == null)
663 return BridgeError("Missing dmApiVersion field!");
664
665 DMApiVersion = parameters.Version;
666 if (DMApiVersion.Major != DMApiConstants.InteropVersion.Major)
667 {
668 apiValidationStatus = ApiValidationStatus.Incompatible;
669 return BridgeError("Incompatible dmApiVersion!");
670 }
671
672 switch (parameters.MinimumSecurityLevel)
673 {
674 case DreamDaemonSecurity.Ultrasafe:
675 apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
676 break;
677 case DreamDaemonSecurity.Safe:
678 apiValidationStatus = ApiValidationStatus.RequiresSafe;
679 break;
680 case DreamDaemonSecurity.Trusted:
681 apiValidationStatus = ApiValidationStatus.RequiresTrusted;
682 break;
683 case null:
684 return BridgeError("Missing minimumSecurityLevel field!");
685 default:
686 return BridgeError("Invalid minimumSecurityLevel!");
687 }
688
689 Logger.LogTrace("ApiValidationStatus set to {apiValidationStatus}", apiValidationStatus);
690
691 response.RuntimeInformation = new RuntimeInformation(
692 chatTrackingContext,
693 ReattachInformation.Dmb,
694 ReattachInformation.RuntimeInformation.ServerVersion,
695 ReattachInformation.RuntimeInformation.InstanceName,
696 ReattachInformation.RuntimeInformation.SecurityLevel,
697 ReattachInformation.RuntimeInformation.Visibility,
698 ReattachInformation.RuntimeInformation.ServerPort,
699 ReattachInformation.RuntimeInformation.ApiValidateOnly);
700
701 // Load custom commands
702 chatTrackingContext.CustomCommands = parameters.CustomCommands;
703 break;
704 case BridgeCommandType.Reboot:
705 if (ClosePortOnReboot)
706 {
707 chatTrackingContext.Active = false;
708 response.NewPort = 0;
709 portClosedForReboot = true;
710 }
711
712 var oldRebootTcs = rebootTcs;
713 rebootTcs = new TaskCompletionSource();
714 oldRebootTcs.SetResult();
715 break;
716 case BridgeCommandType.Chunk:
717 return await ProcessChunk<BridgeParameters, BridgeResponse>(ProcessBridgeCommand, BridgeError, parameters.Chunk, cancellationToken);
718 case null:
719 return BridgeError("Missing commandType!");
720 default:
721 return BridgeError($"commandType {parameters.CommandType} not supported!");
722 }
723
724 return response;
725 }

747 async Task<CombinedTopicResponse> SendTopicRequest(TopicParameters parameters, CancellationToken cancellationToken)
748 {
749 parameters.AccessIdentifier = ReattachInformation.AccessIdentifier;
750
751 var fullCommandString = GenerateQueryString(parameters, out var json);
752 Logger.LogTrace("Topic request: {json}", json);
753 var fullCommandByteCount = Encoding.UTF8.GetByteCount(fullCommandString);
754 if (fullCommandByteCount <= DMApiConstants.MaximumTopicRequestLength)
755 return await SendRawTopic(fullCommandString, cancellationToken);
756
757 var interopChunkingVersion = new Version(5, 6, 0);
758 if (ReattachInformation.Dmb.CompileJob.DMApiVersion < interopChunkingVersion)
759 {
760 Logger.LogWarning(
761 "Cannot send topic request as it is exceeds the single request limit of {limitBytes}B ({actualBytes}B) and requires chunking and the current compile job's interop version must be at least {chunkingVersionRequired}!",
762 DMApiConstants.MaximumTopicRequestLength,
763 fullCommandByteCount,
764 interopChunkingVersion);
765 return null;
766 }
767
768 var payloadId = NextPayloadId;
769
770 // AccessIdentifer is just noise in a chunked request
771 parameters.AccessIdentifier = null;
772 GenerateQueryString(parameters, out json);
773
774 // yes, this straight up ignores unicode, precalculating it is useless when we don't
775 // even know if the UTF8 bytes of the url encoded chunk will fit the window until we do said encoding
776 var fullPayloadSize = (uint)json.Length;
777
778 List<string> chunkQueryStrings = null;
779 for (var chunkCount = 2; chunkQueryStrings == null; ++chunkCount)
780 {
781 var standardChunkSize = fullPayloadSize / chunkCount;
782 var bigChunkSize = standardChunkSize + (fullPayloadSize % chunkCount);
783 if (bigChunkSize > DMApiConstants.MaximumTopicRequestLength)
784 continue;
785
786 chunkQueryStrings = new List<string>();
787 for (var i = 0U; i < chunkCount; ++i)
788 {
789 var startIndex = i * standardChunkSize;
790 var subStringLength = Math.Min(
791 fullPayloadSize - startIndex,
792 i == chunkCount - 1
793 ? bigChunkSize
794 : standardChunkSize);
795 var chunkPayload = json.Substring((int)startIndex, (int)subStringLength);
796
797 var chunk = new ChunkData
798 {
799 Payload = chunkPayload,
800 PayloadId = payloadId,
801 SequenceId = i,
802 TotalChunks = (uint)chunkCount,
803 };
804
805 var chunkParameters = new TopicParameters(chunk)
806 {
807 AccessIdentifier = ReattachInformation.AccessIdentifier,
808 };
809
810 var chunkCommandString = GenerateQueryString(chunkParameters, out _);
811 if (Encoding.UTF8.GetByteCount(chunkCommandString) > DMApiConstants.MaximumTopicRequestLength)
812 {
813 // too long when encoded, need more chunks
814 chunkQueryStrings = null;
815 break;
816 }
817
818 chunkQueryStrings.Add(chunkCommandString);
819 }
820 }
821
822 Logger.LogTrace("Chunking topic request ({totalChunks} total)...", chunkQueryStrings.Count);
823
824 CombinedTopicResponse combinedResponse = null;
825 bool LogRequestIssue(bool possiblyFromCompletedRequest)
826 {
827 if (combinedResponse?.InteropResponse == null || combinedResponse.InteropResponse.ErrorMessage != null)
828 {
829 Logger.LogWarning(
830 "Topic request {chunkingStatus} failed!{potentialRequestError}",
831 possiblyFromCompletedRequest ? "final chunk" : "chunking",
832 combinedResponse?.InteropResponse?.ErrorMessage != null
833 ? $" Request error: {combinedResponse.InteropResponse.ErrorMessage}"
834 : String.Empty);
835 return true;
836 }
837
838 return false;
839 }
840
841 foreach (var chunkCommandString in chunkQueryStrings)
842 {
843 combinedResponse = await SendRawTopic(chunkCommandString, cancellationToken);
844 if (LogRequestIssue(chunkCommandString == chunkQueryStrings.Last()))
845 return null;
846 }
847
848 while ((combinedResponse.InteropResponse.MissingChunks?.Count ?? 0) > 0)
849 {
850 Logger.LogWarning("DD is still missing some chunks of topic request P{payloadId}! Sending missing chunks...", payloadId);
851 var lastIndex = combinedResponse.InteropResponse.MissingChunks.Last();
852 foreach (var missingChunkIndex in combinedResponse.InteropResponse.MissingChunks)
853 {
854 var chunkCommandString = chunkQueryStrings[(int)missingChunkIndex];
855 combinedResponse = await SendRawTopic(chunkCommandString, cancellationToken);
856 if (LogRequestIssue(missingChunkIndex == lastIndex))
857 return null;
858 }
859 }
860
861 return combinedResponse;
862 }
