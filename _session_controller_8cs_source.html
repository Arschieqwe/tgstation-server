329  public async Task<BridgeResponse> ProcessBridgeRequest(BridgeParameters parameters, CancellationToken cancellationToken)
330  {
331  if (parameters == null)
332  throw new ArgumentNullException(nameof(parameters));
333 
334  var response = new BridgeResponse();
335  switch (parameters.CommandType)
336  {
337  case BridgeCommandType.ChatSend:
338  if (parameters.ChatMessage == null)
339  return new BridgeResponse
340  {
341  ErrorMessage = "Missing chatMessage field!"
342  };
343 
344  if (parameters.ChatMessage.ChannelIds == null)
345  return new BridgeResponse
346  {
347  ErrorMessage = "Missing channelIds field in chatMessage!"
348  };
349 
350  if(parameters.ChatMessage.ChannelIds.Any(channelIdString => !UInt64.TryParse(channelIdString, out var _)))
351  return new BridgeResponse
352  {
353  ErrorMessage = "Invalid channelIds in chatMessage!"
354  };
355 
356  if (parameters.ChatMessage.Text == null)
357  return new BridgeResponse
358  {
359  ErrorMessage = "Missing message field in chatMessage!"
360  };
361 
362  await chat.SendMessage(
363  parameters.ChatMessage.Text,
364  parameters.ChatMessage.ChannelIds.Select(UInt64.Parse),
365  cancellationToken).ConfigureAwait(false);
366  break;
367  case BridgeCommandType.Prime:
368  // currently unused, maybe in the future
369  break;
370  case BridgeCommandType.Kill:
371  logger.LogInformation("Bridge requested process termination!");
372  TerminationWasRequested = true;
373  process.Terminate();
374  break;
375  case BridgeCommandType.PortUpdate:
376  lock (synchronizationLock)
377  {
378  if (!parameters.CurrentPort.HasValue)
379  {
381  logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
382  return new BridgeResponse
383  {
384  ErrorMessage = "Missing stringified port as data parameter!"
385  };
386  }
387 
388  var currentPort = parameters.CurrentPort.Value;
389  if (!nextPort.HasValue)
390  reattachInformation.Port = parameters.CurrentPort.Value; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
391  else
392  {
393  // nextPort is ready, tell DD to switch to that
394  // if it fails it'll kill itself
395  response.NewPort = nextPort.Value;
396  reattachInformation.Port = nextPort.Value;
397  nextPort = null;
398 
399  // we'll also get here from SetPort so complete that task
400  var tmpTcs = portAssignmentTcs;
401  portAssignmentTcs = null;
402  tmpTcs.SetResult(true);
403  }
404 
405  portClosedForReboot = false;
406  }
407 
408  break;
409  case BridgeCommandType.Startup:
410  apiValidationStatus = ApiValidationStatus.BadValidationRequest;
411  if (parameters.Version == null)
412  return new BridgeResponse
413  {
414  ErrorMessage = "Missing dmApiVersion field!"
415  };
416 
417  DMApiVersion = parameters.Version;
418  switch (parameters.MinimumSecurityLevel)
419  {
420  case DreamDaemonSecurity.Ultrasafe:
421  apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
422  break;
423  case DreamDaemonSecurity.Safe:
424  apiValidationStatus = ApiValidationStatus.RequiresSafe;
425  break;
426  case DreamDaemonSecurity.Trusted:
427  apiValidationStatus = ApiValidationStatus.RequiresTrusted;
428  break;
429  case null:
430  return new BridgeResponse
431  {
432  ErrorMessage = "Missing minimumSecurityLevel field!"
433  };
434  default:
435  return new BridgeResponse
436  {
437  ErrorMessage = "Invalid minimumSecurityLevel!"
438  };
439  }
440 
441  response.RuntimeInformation = reattachInformation.RuntimeInformation;
442 
443  // Load custom commands
444  chatTrackingContext.CustomCommands = parameters.CustomCommands.ToList();
445  break;
446  case BridgeCommandType.Reboot:
447  if (ClosePortOnReboot)
448  {
449  chatTrackingContext.Active = false;
450  response.NewPort = 0;
451  portClosedForReboot = true;
452  }
453 
454  var oldTcs = rebootTcs;
455  rebootTcs = new TaskCompletionSource<object>();
456  oldTcs.SetResult(null);
457  break;
458  case null:
459  response.ErrorMessage = "Missing commandType!";
460  break;
461  default:
462  response.ErrorMessage = "Requested commandType not supported!";
463  break;
464  }
465 
466  return response;
467  }
