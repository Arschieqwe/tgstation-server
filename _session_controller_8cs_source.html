300  var query = command.Parameters;
301 
302  object content;
303  Action postRespond = null;
304  ushort? overrideResponsePort = null;
305  if (query.TryGetValue(Constants.DMParameterCommand, out var method))
306  {
307  content = new object();
308  switch (method)
309  {
310  case Constants.DMCommandChat:
311  try
312  {
313  var message = JsonConvert.DeserializeObject<Response>(command.RawJson, new JsonSerializerSettings
314  {
315  ContractResolver = new CamelCasePropertyNamesContractResolver()
316  });
317  if (message.ChannelIds == null)
318  throw new InvalidOperationException("Missing ChannelIds field!");
319  if (message.Message == null)
320  throw new InvalidOperationException("Missing Message field!");
321  await chat.SendMessage(message.Message, message.ChannelIds, cancellationToken).ConfigureAwait(false);
322  }
323  catch (Exception e)
324  {
325  logger.LogDebug("Exception while decoding chat message! Exception: {0}", e);
326  goto default;
327  }
328 
329  break;
330  case Constants.DMCommandServerPrimed:
331  // currently unused, maybe in the future
332  break;
333  case Constants.DMCommandEndProcess:
334  TerminationWasRequested = true;
335  process.Terminate();
336  return;
337  case Constants.DMCommandNewPort:
338  lock (this)
339  {
340  if (!query.TryGetValue(Constants.DMParameterData, out var stringPortObject) || !UInt16.TryParse(stringPortObject as string, out var currentPort))
341  {
343  logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
344  content = new ErrorMessage { Message = "Missing stringified port as data parameter!" };
345  break;
346  }
347 
348  if (!nextPort.HasValue)
349  reattachInformation.Port = currentPort; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
350  else
351  {
352  // nextPort is ready, tell DD to switch to that
353  // if it fails it'll kill itself
354  content = new Dictionary<string, ushort> { { Constants.DMParameterData, nextPort.Value } };
355  reattachInformation.Port = nextPort.Value;
356  overrideResponsePort = currentPort;
357  nextPort = null;
358 
359  // we'll also get here from SetPort so complete that task
360  var tmpTcs = portAssignmentTcs;
361  portAssignmentTcs = null;
362  if (tmpTcs != null)
363  postRespond = () => tmpTcs.SetResult(true);
364  }
365 
366  portClosedForReboot = false;
367  }
368 
369  break;
370  case Constants.DMCommandApiValidate:
371  if (!launchSecurityLevel.HasValue)
372  {
373  logger.LogWarning("DreamDaemon requested API validation but no intial security level was passed to the session controller!");
374  apiValidationStatus = ApiValidationStatus.UnaskedValidationRequest;
375  content = new ErrorMessage { Message = "Invalid API validation request!" };
376  break;
377  }
378

381  else

384  case DreamDaemonSecurity.Safe:

386  break;
387  case DreamDaemonSecurity.Ultrasafe:
388  apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
389  break;
390  case DreamDaemonSecurity.Trusted:

395  }

397  break;

400  {
401  chatJsonTrackingContext.Active = false;

412  break;
413  }
