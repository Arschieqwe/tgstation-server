233 public async Task<ISessionController> LaunchNew(
234 IDmbProvider dmbProvider,
235 IByondExecutableLock currentByondLock,
236 DreamDaemonLaunchParameters launchParameters,
237 bool apiValidate,
238 CancellationToken cancellationToken)
239 {
240 logger.LogTrace("Begin session launch...");
241 if (!launchParameters.Port.HasValue)
242 throw new InvalidOperationException("Given port is null!");
243 switch (dmbProvider.CompileJob.MinimumSecurityLevel)
244 {
245 case DreamDaemonSecurity.Ultrasafe:
246 break;
247 case DreamDaemonSecurity.Safe:
248 if (launchParameters.SecurityLevel == DreamDaemonSecurity.Ultrasafe)
249 {
250 logger.LogTrace("Boosting security level to minimum of Safe");
251 launchParameters.SecurityLevel = DreamDaemonSecurity.Safe;
252 }
253
254 break;
255 case DreamDaemonSecurity.Trusted:
256 if (launchParameters.SecurityLevel != DreamDaemonSecurity.Trusted)
257 logger.LogTrace("Boosting security level to minimum of Trusted");
258
259 launchParameters.SecurityLevel = DreamDaemonSecurity.Trusted;
260 break;
261 default:
262 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Invalid DreamDaemonSecurity value: {0}", dmbProvider.CompileJob.MinimumSecurityLevel));
263 }
264
265 // get the byond lock
266 var byondLock = currentByondLock ?? await byond.UseExecutables(Version.Parse(dmbProvider.CompileJob.ByondVersion), cancellationToken);
267 try
268 {
269 logger.LogDebug(
270 "Launching session with CompileJob {compileJobId}...",
271 dmbProvider.CompileJob.Id);
272
273 if (launchParameters.SecurityLevel == DreamDaemonSecurity.Trusted)
274 await byondLock.TrustDmbPath(gameIOManager.ConcatPath(dmbProvider.Directory, dmbProvider.DmbName), cancellationToken);
275
276 PortBindTest(launchParameters.Port.Value);
277 await CheckPagerIsNotRunning(cancellationToken);
278
279 string outputFilePath = null;
280 if (launchParameters.LogOutput.Value)
281 {
282 var now = DateTimeOffset.UtcNow;
283 var dateDirectory = diagnosticsIOManager.ConcatPath(DreamDaemonLogsPath, now.ToString("yyyy-MM-dd", CultureInfo.InvariantCulture));
284 await diagnosticsIOManager.CreateDirectory(dateDirectory, cancellationToken);
285 outputFilePath = diagnosticsIOManager.ResolvePath(
286 diagnosticsIOManager.ConcatPath(
287 dateDirectory,
288 $"dd-utc-{now.ToString("yyyy-MM-dd-HH-mm-ss", CultureInfo.InvariantCulture)}{(apiValidate ? "-dmapi" : String.Empty)}.log"));
289
290 logger.LogInformation("Logging DreamDaemon output to {path}...", outputFilePath);
291 }
292 else if (!byondLock.SupportsCli)
293 outputFilePath = gameIOManager.ConcatPath(dmbProvider.Directory, $"{Guid.NewGuid()}.dd.log");
294
295 var accessIdentifier = cryptographySuite.GetSecureString();
296
297 var byondTopicSender = topicClientFactory.CreateTopicClient(
298 TimeSpan.FromMilliseconds(
299 launchParameters.TopicRequestTimeout.Value));
300
301 if (!apiValidate && dmbProvider.CompileJob.DMApiVersion == null)
302 logger.LogDebug("Session will have no DMAPI support!");
303
304 // launch dd
305 var process = await CreateDreamDaemonProcess(
306 dmbProvider,
307 byondTopicSender,
308 byondLock,
309 launchParameters,
310 accessIdentifier,
311 outputFilePath,
312 apiValidate,
313 cancellationToken);
314
315 try
316 {
317 var chatTrackingContext = chat.CreateTrackingContext();
318
319 try
320 {
321 var runtimeInformation = CreateRuntimeInformation(
322 dmbProvider,
323 chatTrackingContext,
324 launchParameters,
325 apiValidate);
326
327 var reattachInformation = new ReattachInformation(
328 dmbProvider,
329 process,
330 runtimeInformation,
331 accessIdentifier,
332 launchParameters.Port.Value);
333
334 var sessionController = new SessionController(
335 reattachInformation,
336 instance,
337 process,
338 byondLock,
339 byondTopicSender,
340 chatTrackingContext,
341 bridgeRegistrar,
342 chat,
343 assemblyInformationProvider,
344 loggerFactory.CreateLogger<SessionController>(),
345 () => !launchParameters.LogOutput.Value
346 ? LogDDOutput(process, outputFilePath, byondLock.SupportsCli, default) // DCT: None available
347 : Task.CompletedTask,
348 launchParameters.StartupTimeout,
349 false,
350 apiValidate);
351
352 return sessionController;
353 }
354 catch
355 {
356 chatTrackingContext.Dispose();
357 throw;
358 }
359 }
360 catch
361 {
362 await using (process)
363 {
364 process.Terminate();
365 await process.Lifetime;
366 throw;
367 }
368 }
369 }
370 catch
371 {
372 if (currentByondLock == null)
373 byondLock.Dispose();
374 throw;
375 }
376 }

379 public async Task<ISessionController> Reattach(
380 ReattachInformation reattachInformation,
381 CancellationToken cancellationToken)
382 {
383 if (reattachInformation == null)
384 throw new ArgumentNullException(nameof(reattachInformation));
385
386 logger.LogTrace("Begin session reattach...");
387 var byondTopicSender = topicClientFactory.CreateTopicClient(reattachInformation.TopicRequestTimeout);
388 var byondLock = await byond.UseExecutables(Version.Parse(reattachInformation.Dmb.CompileJob.ByondVersion), cancellationToken);
389
390 try
391 {
392 logger.LogDebug(
393 "Attaching to session PID: {0}, CompileJob: {1}...",
394 reattachInformation.ProcessId,
395 reattachInformation.Dmb.CompileJob.Id);
396
397 var process = processExecutor.GetProcess(reattachInformation.ProcessId);
398 if (process == null)
399 return null;
400
401 try
402 {
403 networkPromptReaper.RegisterProcess(process);
404
405 var chatTrackingContext = chat.CreateTrackingContext();
406 try
407 {
408 var runtimeInformation = CreateRuntimeInformation(
409 reattachInformation.Dmb,
410 chatTrackingContext,
411 null,
412 false);
413 reattachInformation.SetRuntimeInformation(runtimeInformation);
414
415 var controller = new SessionController(
416 reattachInformation,
417 instance,
418 process,
419 byondLock,
420 byondTopicSender,
421 chatTrackingContext,
422 bridgeRegistrar,
423 chat,
424 assemblyInformationProvider,
425 loggerFactory.CreateLogger<SessionController>(),
426 () => Task.CompletedTask,
427 null,
428 true,
429 false);
430
431 process = null;
432 byondLock = null;
433 chatTrackingContext = null;
434
435 return controller;
436 }
437 catch
438 {
439 chatTrackingContext.Dispose();
440 throw;
441 }
442 }
443 catch
444 {
445 await process.DisposeAsync();
446 throw;
447 }
448 }
449 catch
450 {
451 byondLock.Dispose();
452 throw;
453 }

468 async Task<IProcess> CreateDreamDaemonProcess(
469 IDmbProvider dmbProvider,
470 ITopicClient byondTopicSender,
471 IByondExecutableLock byondLock,
472 DreamDaemonLaunchParameters launchParameters,
473 string accessIdentifier,
474 string logFilePath,
475 bool apiValidate,
476 CancellationToken cancellationToken)
477 {
478 // set command line options
479 // more sanitization here cause it uses the same scheme
480 var parameters = $"{DMApiConstants.ParamApiVersion}={byondTopicSender.SanitizeString(DMApiConstants.InteropVersion.Semver().ToString())}&{byondTopicSender.SanitizeString(DMApiConstants.ParamServerPort)}={serverPortProvider.HttpApiPort}&{byondTopicSender.SanitizeString(DMApiConstants.ParamAccessIdentifier)}={byondTopicSender.SanitizeString(accessIdentifier)}";
481
482 if (!String.IsNullOrEmpty(launchParameters.AdditionalParameters))
483 parameters = $"{parameters}&{launchParameters.AdditionalParameters}";
484
485 // important to run on all ports to allow port changing
486 var arguments = String.Format(
487 CultureInfo.InvariantCulture,
488 "{0} -port {1} -ports 1-65535 {2}-close -verbose -{3} -{4}{5}{6} -params \"{7}\"",
489 dmbProvider.DmbName,
490 launchParameters.Port.Value,
491 launchParameters.AllowWebClient.Value ? "-webclient " : String.Empty,
492 SecurityWord(launchParameters.SecurityLevel.Value),
493 VisibilityWord(launchParameters.Visibility.Value),
494 !byondLock.SupportsCli
495 ? $" -logself -log {logFilePath}"
496 : !platformIdentifier.IsWindows // Just use stdout on if CLI is supported
497 ? " -logself"
498 : String.Empty, // Windows doesn't output anything to dd.exe if -logself is set?
499 launchParameters.StartProfiler.Value
500 ? " -profile"
501 : String.Empty,
502 parameters);
503
504 var process = await processExecutor.LaunchProcess(
505 byondLock.DreamDaemonPath,
506 dmbProvider.Directory,
507 arguments,
508 logFilePath,
509 byondLock.SupportsCli,
510 true);
511
512 try
513 {
514 if (apiValidate)
515 {
516 if (sessionConfiguration.HighPriorityLiveDreamDaemon)
517 process.AdjustPriority(true);
518 }
519 else if (sessionConfiguration.LowPriorityDeploymentProcesses)
520 process.AdjustPriority(false);
521
522 networkPromptReaper.RegisterProcess(process);
523
524 // If this isnt a staging DD (From a Deployment), fire off an event
525 if (!apiValidate)
526 await eventConsumer.HandleEvent(
527 EventType.DreamDaemonLaunch,
528 new List<string>
529 {
530 process.Id.ToString(CultureInfo.InvariantCulture),
531 },
532 cancellationToken);
533
534 return process;
535 }
536 catch
537 {
538 await using (process)
539 {
540 process.Terminate();
541 await process.Lifetime;
542 throw;
543 }
544 }
545 }

555 async Task LogDDOutput(IProcess process, string outputFilePath, bool cliSupported, CancellationToken cancellationToken)
556 {
557 try
558 {
559 string ddOutput;
560 if (cliSupported)
561 ddOutput = await process.GetCombinedOutput(cancellationToken);
562 else
563 try
564 {
565 var dreamDaemonLogBytes = await gameIOManager.ReadAllBytes(
566 outputFilePath,
567 cancellationToken);
568
569 ddOutput = Encoding.UTF8.GetString(dreamDaemonLogBytes);
570 }
571 finally
572 {
573 try
574 {
575 await gameIOManager.DeleteFile(outputFilePath, cancellationToken);

577 catch (Exception ex)

579 logger.LogWarning(ex, "Failed to delete DreamDaemon log file {outputFilePath}!", outputFilePath);

581 }

583 logger.LogTrace(
584 "DreamDaemon Output:{newLine}{output}",
585 Environment.NewLine,
586 ddOutput);
587 }
588 catch (Exception ex)
589 {
590 logger.LogWarning(ex, "Error reading DreamDaemon output!");

622 async Task CheckPagerIsNotRunning(CancellationToken cancellationToken)
623 {
624 if (!platformIdentifier.IsWindows)
625 return;
626
627 await using var otherProcess = processExecutor.GetProcessByName("byond");
628 if (otherProcess == null)
629 return;
630
631 var otherUsernameTask = otherProcess.GetExecutingUsername(cancellationToken);
632 await using var ourProcess = processExecutor.GetCurrentProcess();
633 var ourUserName = await ourProcess.GetExecutingUsername(cancellationToken);
634 var otherUserName = await otherUsernameTask;
635
636 if (otherUserName.Equals(ourUserName, StringComparison.Ordinal))
637 throw new JobException(ErrorCode.DeploymentPagerRunning);
638 }
