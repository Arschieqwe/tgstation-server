168  var basePath = primaryDirectory ? dmbProvider.PrimaryDirectory : dmbProvider.SecondaryDirectory;
169 
170  switch (dmbProvider.CompileJob.MinimumSecurityLevel)

172  case DreamDaemonSecurity.Ultrasafe:
173  break;
174  case DreamDaemonSecurity.Safe:
175  if (launchParameters.SecurityLevel == DreamDaemonSecurity.Ultrasafe)
176  launchParameters.SecurityLevel = DreamDaemonSecurity.Safe;
177  break;
178  case DreamDaemonSecurity.Trusted:
179  launchParameters.SecurityLevel = DreamDaemonSecurity.Trusted;
180  break;
181  default:
182  throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Invalid DreamDaemonSecurity value: {0}", dmbProvider.CompileJob.MinimumSecurityLevel));
183  }
184 
185  var chatTrackingContext = chat.CreateTrackingContext();
186  try
187  {
188  // get the byond lock
189  var byondLock = currentByondLock ?? await byond.UseExecutables(Version.Parse(dmbProvider.CompileJob.ByondVersion), cancellationToken).ConfigureAwait(false);
190  try
191  {
192  if (launchParameters.SecurityLevel == DreamDaemonSecurity.Trusted)
193  await byondLock.TrustDmbPath(ioManager.ConcatPath(basePath, dmbProvider.DmbName), cancellationToken).ConfigureAwait(false);
194 
195  var accessIdentifier = cryptographySuite.GetSecureString();
196 
197  // set command line options
198  // more sanitization here cause it uses the same scheme
199  var parameters = $"{DMApiConstants.ParamApiVersion}={byondTopicSender.SanitizeString(DMApiConstants.Version.Semver().ToString())}&{byondTopicSender.SanitizeString(DMApiConstants.ParamServerPort)}={serverPortProvider.HttpApiPort}&{byondTopicSender.SanitizeString(DMApiConstants.ParamAccessIdentifier)}={byondTopicSender.SanitizeString(accessIdentifier)}";
200 
201  var visibility = apiValidate ? "invisible" : "public";
202 
203  // important to run on all ports to allow port changing
204  var arguments = String.Format(CultureInfo.InvariantCulture, "{0} -port {1} -ports 1-65535 {2}-close -{3} -{4} -public -params \"{5}\"",

206  primaryPort ? launchParameters.PrimaryPort : launchParameters.SecondaryPort,
