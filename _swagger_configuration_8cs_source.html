213  Tuple<PropertyInfo, string, OpenApiSchema, IDictionary<string, OpenApiSchema>> GetTypeFromKvp(Type currentType, KeyValuePair<string, OpenApiSchema> kvp, IDictionary<string, OpenApiSchema> schemaDictionary)
214  {
215  var propertyInfo = currentType
216  .GetProperties()
217  .Single(x => x.Name.Equals(kvp.Key, StringComparison.OrdinalIgnoreCase));
218 
219  return Tuple.Create(
220  propertyInfo,
221  kvp.Key,
222  kvp.Value,
223  schemaDictionary);
224  }
225 
226  var subSchemaStack = new Stack<Tuple<PropertyInfo, string, OpenApiSchema, IDictionary<string, OpenApiSchema>>>(
227  rootSchema
228  .Properties
229  .Select(
230  x => GetTypeFromKvp(context.Type, x, rootSchema.Properties))
231  .Where(x => x.Item3.Reference == null));
232 
233  while (subSchemaStack.Count > 0)
234  {
235  var tuple = subSchemaStack.Pop();
236  var subSchema = tuple.Item3;
237 
238  var subSchemaPropertyInfo = tuple.Item1;
239 
240  if (subSchema.Properties != null
241  && !subSchemaPropertyInfo
242  .PropertyType
243  .GetInterfaces()
244  .Any(x => x == typeof(IEnumerable)))
245  foreach (var kvp in subSchema.Properties.Where(x => x.Value.Reference == null))
246  subSchemaStack.Push(GetTypeFromKvp(subSchemaPropertyInfo.PropertyType, kvp, subSchema.Properties));
247 
248  var attributes = subSchemaPropertyInfo
249  .GetCustomAttributes();
250  var responsePresence = attributes
251  .OfType<ResponseOptionsAttribute>()
252  .FirstOrDefault()
253  ?.Presence
254  ?? FieldPresence.Required;
255  var requestOptions = attributes
256  .OfType<RequestOptionsAttribute>()
257  .OrderBy(x => x.PutOnly) // Process PUTs last
258  .ToList();
259 
260  if (requestOptions.Any() && requestOptions.All(x => x.Presence == FieldPresence.Ignored && !x.PutOnly))
261  subSchema.ReadOnly = true;
262 
263  var subSchemaId = tuple.Item2;
264  var subSchemaOwningDictionary = tuple.Item4;
265  if (rootResponseSchema)
266  {
267  subSchema.Nullable = responsePresence == FieldPresence.Optional;
268  if (responsePresence == FieldPresence.Ignored)
269  subSchemaOwningDictionary.Remove(subSchemaId);
270  }
271  else if (rootRequestSchema)
272  {
273  subSchema.Nullable = true;
274  var lastOptionWasIgnored = false;
275  foreach (var requestOption in requestOptions)
276  {
277  var validForThisRequest = !requestOption.PutOnly || isPutRequest;
278  if (!validForThisRequest)
279  continue;
280 
281  lastOptionWasIgnored = false;
282  switch (requestOption.Presence)
283  {
284  case FieldPresence.Ignored:
285  lastOptionWasIgnored = true;
286  break;
287  case FieldPresence.Optional:
288  subSchema.Nullable = true;
289  break;
290  case FieldPresence.Required:
291  subSchema.Nullable = false;
292  break;
293  default:
294  throw new InvalidOperationException($"Invalid FieldPresence: {requestOption.Presence}!");
295  }
296  }
297 
298  if (lastOptionWasIgnored)
299  subSchemaOwningDictionary.Remove(subSchemaId);
300  }
301  else if (responsePresence == FieldPresence.Required
302  && requestOptions.All(x => x.Presence == FieldPresence.Required && !x.PutOnly))
303  subSchema.Nullable = subSchemaId.Equals(
304  nameof(TestMergeParameters.TargetCommitSha),
305  StringComparison.OrdinalIgnoreCase)
306  && rootSchemaId == nameof(TestMergeParameters); // special tactics
307 
308  // otherwise, we have to assume it's a shared schema
309  // use what Swagger thinks the nullability is by default
310  }
