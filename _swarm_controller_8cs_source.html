194 public override async Task OnActionExecutionAsync(ActionExecutingContext context, ActionExecutionDelegate next)
195 {
196 using (LogContext.PushProperty("Request", $"{Request.Method} {Request.Path}"))
197 {
198 logger.LogTrace("Swarm request from {0}...", Request.HttpContext.Connection.RemoteIpAddress);
199 if (swarmConfiguration.PrivateKey == null)
200 {
201 logger.LogDebug("Attempted swarm request without private key configured!");
202 await Forbid().ExecuteResultAsync(context);
203 return;
204 }
205
206 if (!(Request.Headers.TryGetValue(SwarmConstants.ApiKeyHeader, out var apiKeyHeaderValues)
207 && apiKeyHeaderValues.Count == 1
208 && apiKeyHeaderValues.First() == swarmConfiguration.PrivateKey))
209 {
210 logger.LogDebug("Unauthorized swarm request!");
211 await Unauthorized().ExecuteResultAsync(context);
212 return;
213 }
214
215 if (!(Request.Headers.TryGetValue(SwarmConstants.RegistrationIdHeader, out var registrationHeaderValues)
216 && registrationHeaderValues.Count == 1
217 && Guid.TryParse(registrationHeaderValues.First(), out var registrationId)))
218 {
219 logger.LogDebug("Swarm request without registration ID!");
220 await BadRequest().ExecuteResultAsync(context);
221 return;
222 }
223
224 // we validate the registration itself on a case-by-case basis
225 if (ModelState?.IsValid == false)
226 {
227 var errors = ModelState
228 .SelectMany(x => x.Value.Errors)
229 .Select(x => x.Exception);
230
231 logger.LogDebug(new AggregateException(errors), "Swarm request model validation failed!");
232 await BadRequest().ExecuteResultAsync(context);
233 return;
234 }
235
236 logger.LogTrace("Starting swarm request processing...");
237 await base.OnActionExecutionAsync(context, next);
238 }
239 }
