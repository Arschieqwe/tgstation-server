1040  async Task HealthCheckLoop(CancellationToken cancellationToken)
1041  {
1042  logger.LogTrace("Starting HealthCheckLoop...");
1043  try
1044  {
1045  var nextForceHealthCheckTask = forceHealthCheckTcs.Task;
1046  while (!cancellationToken.IsCancellationRequested)
1047  {
1048  TimeSpan delay;
1049  if (swarmController)
1050  delay = TimeSpan.FromMinutes(ControllerHealthCheckIntervalMinutes);
1051  else
1052  {
1053  delay = TimeSpan.FromMinutes(NodeHealthCheckIntervalMinutes);
1054  if (lastControllerHealthCheck.HasValue)
1055  {
1056  var recommendedTimeOfNextCheck = lastControllerHealthCheck.Value + delay;
1057 
1058  if (recommendedTimeOfNextCheck > DateTimeOffset.UtcNow)
1059  delay = recommendedTimeOfNextCheck - DateTimeOffset.UtcNow;
1060  }
1061  }
1062 
1063  var delayTask = asyncDelayer.Delay(
1064  delay,
1065  cancellationToken);
1066 
1067  var awakeningTask = Task.WhenAny(
1068  delayTask,
1069  nextForceHealthCheckTask);
1070 
1071  await awakeningTask.ConfigureAwait(false);
1072 
1073  if (nextForceHealthCheckTask.IsCompleted && swarmController)
1074  {
1075  // Intentionally wait a few seconds for the other server to start up before interogating it
1076  logger.LogTrace("Next health check triggering in {0}s...", SecondsToDelayForcedHealthChecks);
1077  await asyncDelayer.Delay(TimeSpan.FromSeconds(SecondsToDelayForcedHealthChecks), cancellationToken);
1078  }
1079  else if (!swarmController && !nextForceHealthCheckTask.IsCompleted)
1080  {
1081  if (!lastControllerHealthCheck.HasValue)
1082  {
1083  logger.LogTrace("Not initially registered with controller, skipping health check.");
1084  continue; // unregistered
1085  }
1086 
1087  if ((DateTimeOffset.UtcNow - lastControllerHealthCheck.Value).TotalMinutes < NodeHealthCheckIntervalMinutes)
1088  {
1089  logger.LogTrace("Controller seems to be active, skipping health check.");
1090  continue;
1091  }
1092  }
1093 
1094  nextForceHealthCheckTask = forceHealthCheckTcs.Task;
1095 
1096  logger.LogTrace("Performing swarm health check...");
1097  try
1098  {
1099  if (swarmController)
1100  await HealthCheckNodes(cancellationToken).ConfigureAwait(false);
1101  else
1102  await HealthCheckController(cancellationToken).ConfigureAwait(false);
1103  }
1104  catch (Exception ex) when (!(ex is OperationCanceledException))
1105  {
1106  logger.LogError(ex, "Health check error!");
1107  }
1108  }
1109  }
1110  catch (OperationCanceledException ex)
1111  {
1112  logger.LogTrace(ex, "Health check loop cancelled!");
1113  }
1114 
1115  logger.LogTrace("Stopped HealthCheckLoop");
1116  }
