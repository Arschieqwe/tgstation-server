335 public async Task<SwarmCommitResult> CommitUpdate(CancellationToken cancellationToken)
336 {
337 if (!SwarmMode)
338 return SwarmCommitResult.ContinueUpdateNonCommitted;
339
340 // wait for the update commit TCS
341 var commitTcsTask = updateCommitTcs?.Task;
342 if (commitTcsTask == null)
343 {
344 logger.LogDebug("Update commit failed, no pending task completion source!");
345 await AbortUpdate(cancellationToken);
346 return SwarmCommitResult.AbortUpdate;
347 }
348
349 logger.LogInformation("Waiting to commit update...");
350 using var httpClient = httpClientFactory.CreateClient();
351 if (!swarmController)
352 {
353 // let the controller know we're ready
354 logger.LogTrace("Sending ready-commit to swarm controller...");
355 using var commitReadyRequest = PrepareSwarmRequest(
356 null,
357 HttpMethod.Post,
358 SwarmConstants.UpdateRoute,
359 null);
360
361 try
362 {
363 using var commitReadyResponse = await httpClient.SendAsync(commitReadyRequest, cancellationToken);
364 commitReadyResponse.EnsureSuccessStatusCode();
365 }
366 catch (Exception ex)
367 {
368 logger.LogWarning(ex, "Unable to send ready-commit to swarm controller!");
369 await AbortUpdate(cancellationToken);
370 return SwarmCommitResult.AbortUpdate;
371 }
372 }
373
374 var timeoutTask = swarmController
375 ? asyncDelayer.Delay(
376 TimeSpan.FromMinutes(UpdateCommitTimeoutMinutes),
377 cancellationToken)
378 : Extensions.TaskExtensions.InfiniteTask().WithToken(cancellationToken);
379
380 var commitTask = Task.WhenAny(commitTcsTask, timeoutTask);
381
382 await commitTask;
383
384 var commitGoAhead = commitTcsTask.IsCompleted
385 && commitTcsTask.Result
386 && updateCommitTcs?.Task == commitTcsTask;
387 if (!commitGoAhead)
388 {
389 logger.LogDebug(
390 "Update commit failed!{maybeTimeout}",
391 timeoutTask.IsCompleted
392 ? " Timed out!"
393 : String.Empty);
394 await AbortUpdate(cancellationToken);
395 return SwarmCommitResult.AbortUpdate;
396 }
397
398 logger.LogTrace("Update commit task complete");
399
400 // on nodes, it means we can go straight ahead
401 if (!swarmController)
402 return SwarmCommitResult.MustCommitUpdate;
403
404 // on the controller, we first need to signal for nodes to go ahead
405 // if anything fails at this point, there's nothing we can do
406 logger.LogDebug("Sending remote commit message to nodes...");
407 async Task SendRemoteCommitUpdate(SwarmServerResponse swarmServer)
408 {
409 using var request = PrepareSwarmRequest(
410 swarmServer,
411 HttpMethod.Post,
412 SwarmConstants.UpdateRoute,
413 null);
414
415 try
416 {
417 // I know using the cancellationToken after this point doesn't seem very sane
418 // It's the token for Ctrl+C on server's console though, so we must respect it
419 using var response = await httpClient.SendAsync(request, cancellationToken);
420 response.EnsureSuccessStatusCode();
421 }
422 catch (Exception ex)
423 {
424 logger.LogCritical(ex, "Failed to send update commit request to node {nodeId}!", swarmServer.Identifier);
425 }
426 }
427
428 Task task;
429 lock (swarmServers)
430 task = Task.WhenAll(
431 swarmServers
432 .Where(x => !x.Controller)
433 .Select(SendRemoteCommitUpdate));
434
435 await task;
436 return SwarmCommitResult.MustCommitUpdate;
437 }

605 public bool RegisterNode(Api.Models.Internal.SwarmServer node, Guid registrationId)
606 {
607 if (node == null)
608 throw new ArgumentNullException(nameof(node));
609
610 if (node.Identifier == null)
611 throw new ArgumentException("Node missing Identifier!", nameof(node));
612
613 if (node.Address == null)
614 throw new ArgumentException("Node missing Address!", nameof(node));
615
616 if (!swarmController)
617 throw new InvalidOperationException("Cannot RegisterNode on swarm node!");
618
619 lock (updateSynchronizationLock)
620 {
621 if (targetUpdateVersion != null)
622 {
623 logger.LogInformation("Not registering node {nodeId} as a distributed update is in progress.", node.Identifier);
624 return false;
625 }
626
627 lock (swarmServers)
628 {
629 if (registrationIds.Any(x => x.Value == registrationId))
630 {
631 var preExistingRegistrationKvp = registrationIds.FirstOrDefault(x => x.Value == registrationId);
632 if (preExistingRegistrationKvp.Key == node.Identifier)
633 {
634 logger.LogWarning("Node {nodeId} has already registered!", node.Identifier);
635 return true;
636 }
637
638 logger.LogWarning(
639 "Registration ID collision! Node {nodeId} tried to register with {otherNodeId}'s registration ID: {registrationId}",
640 node.Identifier,
641 preExistingRegistrationKvp.Key,
642 registrationId);
643 return false;
644 }
645
646 if (registrationIds.TryGetValue(node.Identifier, out var oldRegistration))
647 {
648 logger.LogInformation("Node {nodeId} is re-registering without first unregistering. Indicative of restart.", node.Identifier);
649 swarmServers.RemoveAll(x => x.Identifier == node.Identifier);
650 registrationIds.Remove(node.Identifier);
651 }
652
653 swarmServers.Add(new SwarmServerResponse
654 {
655 Address = node.Address,
656 Identifier = node.Identifier,
657 Controller = false,
658 });
659 registrationIds.Add(node.Identifier, registrationId);
660 }
661 }
662
663 logger.LogInformation("Registered node {nodeId} ({nodeIP}) with ID {registrationId}", node.Identifier, node.Address, registrationId);
664 MarkServersDirty();
665 return true;
666 }

669 public async Task<bool> RemoteCommitRecieved(Guid registrationId, CancellationToken cancellationToken)
670 {
671 if (!swarmController)
672 {
673 logger.LogDebug("Received remote commit go ahead");
674 var commitTcs = updateCommitTcs;
675 commitTcs?.TrySetResult(true);
676 return commitTcs != null;
677 }
678
679 var nodeIdentifier = NodeIdentifierFromRegistration(registrationId);
680 if (nodeIdentifier == null)
681 {
682 // Something fucky is happening, take no chances.
683 logger.LogDebug("Aborting update due to unforseen circumstances!");
684 await AbortUpdate(cancellationToken);
685 return false;
686 }
687
688 var nodeList = nodesThatNeedToBeReadyToCommit;
689 if (nodeList == null)
690 {

692 return false;
693 }
694
695 logger.LogDebug("Node {nodeId} is ready to commit.", nodeIdentifier);
696 lock (nodeList)
697 {
698 nodeList.Remove(nodeIdentifier);
699 if (nodeList.Count == 0)
700 {
701 logger.LogTrace("All nodes ready, update commit is a go once controller is ready");
702 var commitTcs = updateCommitTcs;
703 return commitTcs?.TrySetResult(true) == true;
704 }
705 }
706
707 return true;

745 async Task<bool> PrepareUpdateImpl(Version version, bool initiator, CancellationToken cancellationToken)
746 {
747 if (version == null)
748 throw new ArgumentNullException(nameof(version));
749
750 if (!SwarmMode)
751 return true;
752
753 logger.LogTrace("PrepareUpdateImpl {version}...", version);
754
755 if (version == targetUpdateVersion)
756 {
757 logger.LogDebug("Prepare update short circuit!");
758 return true;

761 using var httpClient = httpClientFactory.CreateClient();
762 async Task<bool> RemotePrepareUpdate(SwarmServerResponse swarmServer)
763 {
764 using var request = PrepareSwarmRequest(
765 swarmServer,
766 HttpMethod.Put,
767 SwarmConstants.UpdateRoute,
768 new SwarmUpdateRequest
769 {
770 UpdateVersion = version,
771 });
772
773 using var response = await httpClient.SendAsync(request, cancellationToken);
774 return response.IsSuccessStatusCode;
775 }
776
777 var shouldAbort = false;
778 try
779 {
780 lock (updateSynchronizationLock)
781 {
782 if (targetUpdateVersion == version)
783 {
784 logger.LogTrace("PrepareUpdateFromController early out, already prepared!");
785 return true;
786 }
787
788 if (targetUpdateVersion != null)
789 {
790 logger.LogWarning("Aborting update preparation, version {targetUpdateVersion} already prepared!", targetUpdateVersion);
791 shouldAbort = true;
792 return false;
793 }
794
795 targetUpdateVersion = version;
796 }
797
798 if (!swarmController && initiator)
799 {
800 logger.LogInformation("Forwarding update request to swarm controller...");
801 var result = await RemotePrepareUpdate(null);
802 if (result)
803 updateCommitTcs = new TaskCompletionSource<bool>();
804
805 return result;
806 }
807
808 if (!initiator)
809 {
810 logger.LogTrace("Beginning local update process...");
811 updateCommitTcs = new TaskCompletionSource<bool>();
812 var updateApplyResult = await serverUpdater.BeginUpdate(
813 this,
814 version,
815 cancellationToken);
816 if (updateApplyResult != ServerUpdateResult.Started)
817 {
818 logger.LogWarning("Failed to prepare update! Result: {serverUpdateResult}", updateApplyResult);
819 shouldAbort = true;
820 return false;
821 }
822 }
823 else
824 {
825 logger.LogTrace("No need to re-initiate update as it originated here on the swarm controller");
826 updateCommitTcs = new TaskCompletionSource<bool>();
827 }
828
829 logger.LogDebug("Local node prepared for update to version {version}", version);
830 }
831 catch (Exception ex)
832 {
833 logger.LogWarning(ex, "Failed to prepare update!");
834 shouldAbort = true;
835 return false;
836 }
837 finally
838 {
839 if (shouldAbort)
840 await AbortUpdate(cancellationToken);
841 }
842
843 if (!swarmController)
844 return true;
845
846 bool abortUpdate = false;
847 try
848 {
849 logger.LogInformation("Sending remote prepare to nodes...");
850 List<Task<bool>> tasks;
851 lock (swarmServers)
852 {
853 nodesThatNeedToBeReadyToCommit = new List<string>(
854 swarmServers
855 .Where(x => !x.Controller)
856 .Select(x => x.Identifier));
857
858 if (nodesThatNeedToBeReadyToCommit.Count < swarmConfiguration.UpdateRequiredNodeCount)
859 {
860 logger.LogWarning(
861 "Aborting update, controller expects to be in sync with {requiredNodeCount} nodes but currently only has {currentNodeCount}!",
862 swarmConfiguration.UpdateRequiredNodeCount,
863 nodesThatNeedToBeReadyToCommit.Count);
864 abortUpdate = true;
865 return false;
866 }
867
868 if (nodesThatNeedToBeReadyToCommit.Count == 0)
869 {
870 logger.LogDebug("Controller has no nodes, setting commit-ready.");
871 var commitTcs = updateCommitTcs;
872 commitTcs?.TrySetResult(true);
873 if (commitTcs != null)
874 return true;
875
876 logger.LogDebug("Update appears to have been aborted");
877 return false;
878 }
879
880 tasks = swarmServers
881 .Where(x => !x.Controller)
882 .Select(RemotePrepareUpdate)
883 .ToList();
884 }
885
886 await Task.WhenAll(tasks);
887
888 // if all succeeds...
889 if (tasks.All(x => x.Result))
890 {
891 logger.LogInformation("Distributed prepare for update to version {version} complete.", version);
892 return true;
893 }
894
895 abortUpdate = true;
896 logger.LogDebug("Distrubuted prepare failed!");
897 }
898 catch (Exception ex)
899 {
900 logger.LogWarning(ex, "Error remotely preparing updates!");
901 }
902 finally
903 {
904 if (abortUpdate)
905 await AbortUpdate(cancellationToken);
906 }
907
908 return false;
909 }

1230 async Task HealthCheckLoop(CancellationToken cancellationToken)
1231 {
1232 logger.LogTrace("Starting HealthCheckLoop...");
1233 try
1234 {
1235 var nextForceHealthCheckTask = forceHealthCheckTcs.Task;
1236 while (!cancellationToken.IsCancellationRequested)
1237 {
1238 TimeSpan delay;
1239 if (swarmController)
1240 delay = TimeSpan.FromMinutes(ControllerHealthCheckIntervalMinutes);
1241 else
1242 {
1243 delay = TimeSpan.FromMinutes(NodeHealthCheckIntervalMinutes);
1244 if (lastControllerHealthCheck.HasValue)
1245 {
1246 var recommendedTimeOfNextCheck = lastControllerHealthCheck.Value + delay;
1247
1248 if (recommendedTimeOfNextCheck > DateTimeOffset.UtcNow)
1249 delay = recommendedTimeOfNextCheck - DateTimeOffset.UtcNow;
1250 }
1251 }
1252
1253 var delayTask = asyncDelayer.Delay(
1254 delay,
1255 cancellationToken);
1256
1257 var awakeningTask = Task.WhenAny(
1258 delayTask,
1259 nextForceHealthCheckTask);
1260
1261 await awakeningTask;
1262
1263 if (nextForceHealthCheckTask.IsCompleted && swarmController)
1264 {
1265 // Intentionally wait a few seconds for the other server to start up before interogating it
1266 logger.LogTrace("Next health check triggering in {delaySeconds}s...", SecondsToDelayForcedHealthChecks);
1267 await asyncDelayer.Delay(TimeSpan.FromSeconds(SecondsToDelayForcedHealthChecks), cancellationToken);
1268 }
1269 else if (!swarmController && !nextForceHealthCheckTask.IsCompleted)
1270 {
1271 if (!lastControllerHealthCheck.HasValue)
1272 {
1273 logger.LogTrace("Not initially registered with controller, skipping health check.");
1274 continue; // unregistered
1275 }
1276
1277 if ((DateTimeOffset.UtcNow - lastControllerHealthCheck.Value).TotalMinutes < NodeHealthCheckIntervalMinutes)
1278 {
1279 logger.LogTrace("Controller seems to be active, skipping health check.");
1280 continue;
1281 }
1282 }
1283
1284 nextForceHealthCheckTask = forceHealthCheckTcs.Task;
1285
1286 logger.LogTrace("Performing swarm health check...");
1287 try
1288 {
1289 if (swarmController)
1290 await HealthCheckNodes(cancellationToken);
1291 else
1292 await HealthCheckController(cancellationToken);
1293 }
1294 catch (Exception ex) when (ex is not OperationCanceledException)
1295 {
1296 logger.LogError(ex, "Health check error!");
1297 }
1298 }
1299 }
1300 catch (OperationCanceledException ex)
1301 {
1302 logger.LogTrace(ex, "Health check loop cancelled!");
1303 }
1304
1305 logger.LogTrace("Stopped HealthCheckLoop");
1306 }
