163  {
164  logger.LogTrace("Verifying DMAPI...");
165  var launchParameters = new DreamDaemonLaunchParameters
166  {
167  AllowWebClient = false,
168  PrimaryPort = portToUse,
169  SecurityLevel = securityLevel,
170  StartupTimeout = timeout
171  };
172 
173  var dirA = ioManager.ConcatPath(job.DirectoryName.ToString(), ADirectoryName);
174 
175  job.MinimumSecurityLevel = securityLevel; // needed for the TempDmbProvider
176  var timeoutAt = DateTimeOffset.Now.AddSeconds(timeout);
177 
178  using (var provider = new TemporaryDmbProvider(ioManager.ResolvePath(dirA), String.Concat(job.DmeName, DmbExtension), job))
179  using (var controller = await sessionControllerFactory.LaunchNew(launchParameters, provider, byondLock, true, true, true, cancellationToken).ConfigureAwait(false))
180  {
181  var launchResult = await controller.LaunchResult.ConfigureAwait(false);
182 
183  var now = DateTimeOffset.Now;
184  if (now < timeoutAt && launchResult.StartupTime.HasValue)
185  {
186  var timeoutTask = Task.Delay(timeoutAt - now, cancellationToken);
187 
188  await Task.WhenAny(controller.Lifetime, timeoutTask).ConfigureAwait(false);
189  cancellationToken.ThrowIfCancellationRequested();
190  }
191 
192  if (controller.Lifetime.IsCompleted)
193  {
194  var validationStatus = controller.ApiValidationStatus;
195  logger.LogTrace("API validation status: {0}", validationStatus);
196  switch (validationStatus)
197  {
198  case ApiValidationStatus.RequiresUltrasafe:
199  job.MinimumSecurityLevel = DreamDaemonSecurity.Ultrasafe;
200  return;
201  case ApiValidationStatus.RequiresSafe:
202  if (securityLevel == DreamDaemonSecurity.Ultrasafe)
203  throw new JobException("This game must be run with at least the 'Safe' DreamDaemon security level!");
204  job.MinimumSecurityLevel = DreamDaemonSecurity.Safe;
205  return;
206  case ApiValidationStatus.RequiresTrusted:
207  if (securityLevel != DreamDaemonSecurity.Trusted)
208  throw new JobException("This game must be run with at least the 'Trusted' DreamDaemon security level!");
209  job.MinimumSecurityLevel = DreamDaemonSecurity.Trusted;
210  return;
211  case ApiValidationStatus.NeverValidated:
212  break;
213  case ApiValidationStatus.BadValidationRequest:
214  throw new JobException("Recieved an unrecognized API validation request from DreamDaemon!");
215  case ApiValidationStatus.UnaskedValidationRequest:
216  default:
217  throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Session controller returned unexpected ApiValidationStatus: {0}", validationStatus));
218  }
219  }
220 
221  throw new JobException("DMAPI validation timed out!");
222  }

385

388 
389  // determine the dme
390  if (job.DmeName == null)
391  {
392  logger.LogTrace("Searching for available .dmes...");
393  var foundPaths = await ioManager.GetFilesWithExtension(dirA, DmeExtension, cancellationToken).ConfigureAwait(false);
394  var foundPath = foundPaths.FirstOrDefault();
395  if (foundPath == default)
396  throw new JobException("Unable to find any .dme!");
397  var dmeWithExtension = ioManager.GetFileName(foundPath);
398  job.DmeName = dmeWithExtension.Substring(0, dmeWithExtension.Length - DmeExtension.Length - 1);
399  }
400  else
401  {
402  var targetDme = ioManager.ConcatPath(dirA, String.Join('.', job.DmeName, DmeExtension));
403  var targetDmeExists = await ioManager.FileExists(targetDme, cancellationToken).ConfigureAwait(false);
404  if (!targetDmeExists)
405  throw new JobException("Unable to locate specified .dme!");
406  }
407 
408  logger.LogDebug("Selected {0}.dme for compilation!", job.DmeName);
409 
410  await ModifyDme(job, cancellationToken).ConfigureAwait(false);
411 
412  // run compiler
413  var exitCode = await RunDreamMaker(byondLock.DreamMakerPath, job, cancellationToken).ConfigureAwait(false);
414 
415  // verify api
416  try
417  {
418  if (exitCode != 0)

428  }
429

431  await eventConsumer.HandleEvent(EventType.CompileComplete, new List<string> { resolvedADirectory }, cancellationToken).ConfigureAwait(false);
432 
433  logger.LogTrace("Duplicating compiled game...");
434 
435  // duplicate the dmb et al
436  await ioManager.CopyDirectory(dirA, dirB, null, cancellationToken).ConfigureAwait(false);
437 
438  logger.LogTrace("Applying static game file symlinks...");
439 
440  // symlink in the static data
441  var symATask = configuration.SymlinkStaticFilesTo(resolvedADirectory, cancellationToken);
442  var symBTask = configuration.SymlinkStaticFilesTo(ioManager.ResolvePath(dirB), cancellationToken);
443 
444  await Task.WhenAll(symATask, symBTask).ConfigureAwait(false);
445 
446  await chat.SendUpdateMessage(String.Format(CultureInfo.InvariantCulture, "Deployment complete!{0}", watchdog.Running ? " Changes will be applied on next server reboot." : String.Empty), cancellationToken).ConfigureAwait(false);
447 
448  logger.LogDebug("Compile complete!");
449  }
450  catch (Exception e)
451  {
452  await CleanupFailedCompile(job, e is OperationCanceledException, cancellationToken).ConfigureAwait(false);
453  throw;
454  }
