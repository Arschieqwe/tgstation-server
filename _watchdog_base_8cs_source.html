242  protected async Task LaunchImplNoLock(bool startMonitor, bool announce, WatchdogReattachInformation reattachInfo, CancellationToken cancellationToken)
243  {
244  Logger.LogTrace("Begin LaunchImplNoLock");
245 
246  if (Running)
247  throw new JobException(ErrorCode.WatchdogRunning);
248 
249  if (!DmbFactory.DmbAvailable)
250  throw new JobException(ErrorCode.WatchdogCompileJobCorrupted);
251 
252  // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
253  Task chatTask;
254  if (startMonitor && await StopMonitor().ConfigureAwait(false))
255  chatTask = Chat.SendWatchdogMessage("Automatic retry sequence cancelled by manual launch. Restarting...", cancellationToken);
256  else if (announce)
257  chatTask = Chat.SendWatchdogMessage(reattachInfo == null ? "Starting..." : "Reattaching...", cancellationToken); // simple announce
258  else
259  chatTask = Task.CompletedTask; // no announce
260 
261  // since neither server is running, this is safe to do
262  LastLaunchParameters = ActiveLaunchParameters;
263 
264  // for when we call ourself and want to not catch thrown exceptions
265  var ignoreNestedException = false;
266  try
267  {
268  await InitControllers(() => ignoreNestedException = true, chatTask, reattachInfo, cancellationToken).ConfigureAwait(false);
269  await chatTask.ConfigureAwait(false);
270 
271  Logger.LogInformation("Launched servers successfully");
272  Running = true;
273 
274  if (startMonitor)
275  {
276  StartMonitor();
277  }
278  }
279  catch (Exception e)
280  {
281  // don't try to send chat tasks or warning logs if were suppressing exceptions or cancelled
282  if (!ignoreNestedException && !cancellationToken.IsCancellationRequested)
283  {
284  var originalChatTask = chatTask;
285  async Task ChainChatTaskWithErrorMessage()
286  {
287  await originalChatTask.ConfigureAwait(false);
288  await Chat.SendWatchdogMessage("Startup failed!", cancellationToken).ConfigureAwait(false);
289  }
290 
291  chatTask = ChainChatTaskWithErrorMessage();

293  }
294 
295  throw;
296  }
297  finally
298  {
299  // finish the chat task that's in flight
300  try
301  {
302  await chatTask.ConfigureAwait(false);
303  }
304  catch (OperationCanceledException) { }
305  }
306  }
