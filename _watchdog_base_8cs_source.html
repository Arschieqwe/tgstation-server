254  protected async Task LaunchImplNoLock(bool startMonitor, bool announce, WatchdogReattachInformation reattachInfo, CancellationToken cancellationToken)
255  {
256  Logger.LogTrace("Begin LaunchNoLock");
257 
258  if (Running)
259  throw new JobException("Watchdog already running!");
260 
261  if (!DmbFactory.DmbAvailable)
262  throw new JobException("Corrupted compilation, please redeploy!");
263 
264  // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
265  Task chatTask;
266  if (startMonitor && await StopMonitor().ConfigureAwait(false))
267  chatTask = Chat.SendWatchdogMessage("Automatic retry sequence cancelled by manual launch. Restarting...", cancellationToken);
268  else if (announce)
269  chatTask = Chat.SendWatchdogMessage(reattachInfo == null ? "Starting..." : "Reattaching...", cancellationToken); // simple announce
270  else
271  chatTask = Task.CompletedTask; // no announce
272 
273  // since neither server is running, this is safe to do
274  LastLaunchParameters = ActiveLaunchParameters;
275 
276  // for when we call ourself and want to not catch thrown exceptions
277  var ignoreNestedException = false;
278  try
279  {
280  await InitControllers(() => ignoreNestedException = true, chatTask, reattachInfo, cancellationToken).ConfigureAwait(false);
281  await chatTask.ConfigureAwait(false);
282 
283  Logger.LogInformation("Launched servers successfully");
284  Running = true;
285 
286  if (startMonitor)
287  {
288  StartMonitor();
289  }
290  }
291  catch (Exception e)
292  {
293  // don't try to send chat tasks or warning logs if were suppressing exceptions or cancelled
294  if (!ignoreNestedException && !cancellationToken.IsCancellationRequested)
295  {
296  var originalChatTask = chatTask;
297  async Task ChainChatTaskWithErrorMessage()
298  {
299  await originalChatTask.ConfigureAwait(false);
300  await Chat.SendWatchdogMessage("Startup failed!", cancellationToken).ConfigureAwait(false);
301  }
302 
303  chatTask = ChainChatTaskWithErrorMessage();

305  }
306 
307  throw;
308  }
309  finally
310  {
311  // finish the chat task that's in flight
312  try
313  {
314  await chatTask.ConfigureAwait(false);
315  }
316  catch (OperationCanceledException) { }
317  }
