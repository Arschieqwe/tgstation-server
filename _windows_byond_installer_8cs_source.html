139 async Task InstallDirectX(string path, CancellationToken cancellationToken)
140 {
141 using var lockContext = await SemaphoreSlimContext.Lock(semaphore, cancellationToken);
142 if (installedDirectX)
143 {
144 Logger.LogTrace("DirectX already installed.");
145 return;
146 }
147
148 Logger.LogTrace("Installing DirectX redistributable...");
149
150 // always install it, it's pretty fast and will do better redundancy checking than us
151 var rbdx = IOManager.ConcatPath(path, ByondDXDir);
152
153 try
154 {
155 // noShellExecute because we aren't doing runas shennanigans
156 using var directXInstaller = processExecutor.LaunchProcess(
157 IOManager.ConcatPath(rbdx, "DXSETUP.exe"),
158 rbdx,
159 "/silent",
160 noShellExecute: true);
161
162 int exitCode;
163 using (cancellationToken.Register(() => directXInstaller.Terminate()))
164 exitCode = await directXInstaller.Lifetime;
165 cancellationToken.ThrowIfCancellationRequested();
166
167 if (exitCode != 0)
168 throw new JobException(ErrorCode.ByondDirectXInstallFail, new JobException($"Invalid exit code: {exitCode}"));
169 installedDirectX = true;
170 }
171 catch (Exception e)
172 {
173 throw new JobException(ErrorCode.ByondDirectXInstallFail, e);
174 }
